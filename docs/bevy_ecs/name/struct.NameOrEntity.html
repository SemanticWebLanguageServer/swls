Expand description

Convenient query for giving a human friendly name to an entity.

fn increment_score(mut scores: Query<(NameOrEntity, &mut Score)>) {
    for (name, mut score) in &mut scores {
        score.0 += 1.0;
        if score.0.is_nan() {
            log::error!("Score for {name} is invalid");
        }
    }
}
