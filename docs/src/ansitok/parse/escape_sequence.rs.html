5/// An ANSI Escape Sequence.
6///
7/// You can find some specification on
8///
9/// - [wiki](https://en.wikipedia.org/wiki/ANSI_escape_code)
10/// - [VT51](https://web.archive.org/web/20090227051140/http://ascii-table.com/ansi-escape-sequences-vt-100.php)
11#[derive(Debug, Eq, PartialEq, PartialOrd, Ord, Clone, Copy, Hash)]
12#[non_exhaustive]
13pub enum EscapeCode<'a> {
14    /// A move cursor backward.
15    ///
16    /// Moves the cursor n (default 1) cells backwards.
17    CursorBackward(u32),
18    /// A cursor down.
19    ///
20    /// Moves the cursor n (default 1) cells down.
21    CursorDown(u32),
22    /// A move cursor forward.
23    ///
24    /// Moves the cursor n (default 1) cells forward.
25    CursorForward(u32),
26    /// A cursor position.
27    ///
28    /// The values are 1-based, and default to 1 (top left corner) if omitted.
29    CursorPos(u32, u32),
30    /// A restore of current cursor position/state.
31    CursorRestore,
32    /// A save of current cursor position/state.
33    CursorSave,
34    /// Set cursor key to application
35    CursorToApp,
36    /// A cursor up.
37    ///
38    /// Moves the cursor n (default 1) cells up.
39    CursorUp(u32),
40    /// Erase in Display.
41    EraseDisplay,
42    /// Erase in Display.
43    EraseLine,
44    /// A ESC sequence.
45    Escape,
46    /// Hide the cursor.
47    HideCursor,
48    /// Reset auto repeat.
49    ResetAutoRepeat,
50    /// Reset auto wrap.
51    ResetAutoWrap,
52    /// Reset interlacin.
53    ResetInterlacing,
54    /// Erase in Display.
55    ResetMode(u8),
56    /// Select Graphic Rendition (SGR), sets display attributes.
57    SelectGraphicRendition(&'a str),
58    /// Set alternate keypad.
59    SetAlternateKeypad,
60    /// Set auto repeat.
61    SetAutoRepeat,
62    /// Set auto wrap.
63    SetAutoWrap,
64    /// Set number of columns to 132
65    SetCol132,
66    /// Set number of columns to 80
67    SetCol80,
68    /// Set cursor key to cursor.
69    SetCursorKeyToCursor,
70    /// Set G0 alt char ROM and spec. graphics.
71    SetG0AltAndSpecialGraph,
72    /// Set G0 alternate character ROM.
73    SetG0AlternateChar,
74    /// Set G0 special chars. & line set.
75    SetG0SpecialChars,
76    /// Set G1 alt char ROM and spec. graphics.
77    SetG1AltAndSpecialGraph,
78    /// Set G1 alternate character ROM.
79    SetG1AlternateChar,
80    /// Set G1 special chars. & line set.
81    SetG1SpecialChars,
82    /// Set interlacing.
83    SetInterlacing,
84    /// Set jump scrolling.
85    SetJumpScrolling,
86    /// Set line feed mode.
87    SetLineFeedMode,
88    /// Erase in Display.
89    SetMode(u8),
90    /// Set new line mode.
91    SetNewLineMode,
92    /// Set normal video.
93    SetNormalVideo,
94    /// Set numeric keypad.
95    SetNumericKeypad,
96    /// Set origin absolute.
97    SetOriginAbsolute,
98    /// Set origin relative.
99    SetOriginRelative,
100    /// Set reverse video.
101    SetReverseVideo,
102    /// Set single shift 2.
103    SetSingleShift2,
104    /// Set single shift 3.
105    SetSingleShift3,
106    /// Set smooth scroll.
107    SetSmoothScroll,
108    /// Set top and bottom lines of a window.
109    SetTopAndBottom(u32, u32),
110    /// Set United Kingdom G0 character set.
111    SetUKG0,
112    /// Set United Kingdom G1 character set.
113    SetUKG1,
114    /// Set United States G0 character set.
115    SetUSG0,
116    /// Set United States G1 character set.
117    SetUSG1,

119    SetVT52,
120    /// Show the cursor.
121    ShowCursor,

124impl EscapeCode<'_> {
125    /// Parse an escape code.
126    /// returns None if the sequence is not supported or it can't be parsed.
127    pub fn parse(text: &str) -> Option<EscapeCode<'_>> {
128        let (_, seq) = parse_escape_sequence(text).ok()?;
129        Some(seq)
130    }
131}

133impl Display for EscapeCode<'_> {
134    fn fmt(&self, formatter: &mut Formatter) -> fmt::Result {
135        write!(formatter, "\u{1b}")?;
136
137        use EscapeCode::*;
138        match self {
139            Escape => write!(formatter, "\u{1b}"),
140            CursorPos(line, col) => write!(formatter, "[{};{}H", line, col),
141            CursorUp(amt) => write!(formatter, "[{}A", amt),
142            CursorDown(amt) => write!(formatter, "[{}B", amt),
143            CursorForward(amt) => write!(formatter, "[{}C", amt),
144            CursorBackward(amt) => write!(formatter, "[{}D", amt),
145            CursorSave => write!(formatter, "[s"),
146            CursorRestore => write!(formatter, "[u"),
147            EraseDisplay => write!(formatter, "[2J"),
148            EraseLine => write!(formatter, "[K"),
149            SelectGraphicRendition(mode) => write!(formatter, "[{}m", mode),
150            SetMode(mode) => write!(formatter, "[={}h", mode),
151            ResetMode(mode) => write!(formatter, "[={}l", mode),
152            ShowCursor => write!(formatter, "[?25h"),
153            HideCursor => write!(formatter, "[?25l"),
154            CursorToApp => write!(formatter, "[?1h"),
155            SetNewLineMode => write!(formatter, "[20h"),
156            SetCol132 => write!(formatter, "[?3h"),
157            SetSmoothScroll => write!(formatter, "[?4h"),
158            SetReverseVideo => write!(formatter, "[?5h"),
159            SetOriginRelative => write!(formatter, "[?6h"),
160            SetAutoWrap => write!(formatter, "[?7h"),
161            SetAutoRepeat => write!(formatter, "[?8h"),
162            SetInterlacing => write!(formatter, "[?9h"),
163            SetLineFeedMode => write!(formatter, "[20l"),
164            SetCursorKeyToCursor => write!(formatter, "[?1l"),
165            SetVT52 => write!(formatter, "[?2l"),
166            SetCol80 => write!(formatter, "[?3l"),
167            SetJumpScrolling => write!(formatter, "[?4l"),
168            SetNormalVideo => write!(formatter, "[?5l"),
169            SetOriginAbsolute => write!(formatter, "[?6l"),
170            ResetAutoWrap => write!(formatter, "[?7l"),
171            ResetAutoRepeat => write!(formatter, "[?8l"),
172            ResetInterlacing => write!(formatter, "[?9l"),
173            SetAlternateKeypad => write!(formatter, "="),
174            SetNumericKeypad => write!(formatter, ">"),
175            SetUKG0 => write!(formatter, "(A"),
176            SetUKG1 => write!(formatter, ")A"),
177            SetUSG0 => write!(formatter, "(B"),
178            SetUSG1 => write!(formatter, ")B"),
179            SetG0SpecialChars => write!(formatter, "(0"),
180            SetG1SpecialChars => write!(formatter, ")0"),
181            SetG0AlternateChar => write!(formatter, "(1"),
182            SetG1AlternateChar => write!(formatter, ")1"),
183            SetG0AltAndSpecialGraph => write!(formatter, "(2"),
184            SetG1AltAndSpecialGraph => write!(formatter, ")2"),
185            SetSingleShift2 => write!(formatter, "N"),
186            SetSingleShift3 => write!(formatter, "O"),
187            SetTopAndBottom(x, y) => write!(formatter, "{};{}r", x, y),
188        }
189    }
190}
