71impl TestClient {
72    pub async fn await_futures<F: FnMut()>(&self, mut tick: F) {
73        tick();
74        while self.tasks_running.load(Ordering::Relaxed) != 0 {
75            self.executor.tick().await;
76            tick();
77        }
78    }

99struct Sendable<T>(pub T);
100
101// Safety: WebAssembly will only ever run in a single-threaded context.
102unsafe impl<T> Send for Sendable<T> {}
103impl<O, T> Future for Sendable<T>
104where
105    T: Future<Output = O>,
106{
107    type Output = O;
108
109    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
110        // Safely access the inner future
111        let inner = unsafe { self.map_unchecked_mut(|s| &mut s.0) };
112        inner.poll(cx)
113    }
114}

116impl ClientSync for TestClient {
117    fn spawn<F: std::future::Future<Output = ()> + 'static>(&self, fut: F) {
118        self.tasks_running.fetch_add(1, Ordering::AcqRel);
119        let tr = self.tasks_running.clone();
120        let fut = async move {
121            fut.await;
122            tr.fetch_sub(1, Ordering::AcqRel);
123        };
124        self.executor.spawn(Sendable(fut)).detach();
125    }
126
127    fn fetch(
128        &self,
129        url: &str,
130        _headers: &std::collections::HashMap<String, String>,
131    ) -> std::pin::Pin<
132        Box<dyn Send + std::future::Future<Output = Result<lsp_core::client::Resp, String>>>,
133    > {
134        let body = self.locations.get(url).cloned();
135        Sendable(async move {
136            let mut headers = Vec::new();
137            async_std::task::sleep(Duration::from_millis(200)).await;
138            headers.push(("Content-Type".to_string(), "text/turtle".to_string()));
139            let status = body.is_some().then_some(200).unwrap_or(404);
140            Ok(Resp {
141                headers,
142                body: body.unwrap_or_default(),
143                status,
144            })
145        })
146        .boxed()
147    }
148}

150pub fn setup_world(
151    client: TestClient,
152    f: impl FnOnce(&mut World) -> (),
153) -> (World, UnboundedReceiver<DiagnosticItem>) {
154    let mut world = World::new();
155    setup_schedule_labels::<TestClient>(&mut world);
156
157    let (tx, rx) = unbounded();
158    world.insert_resource(CommandSender(tx));
159    world.insert_resource(CommandReceiver(rx));
160    world.insert_resource(client);
161
162    let cache = Cache::None(NoCache);
163    let helper = LovHelper::from_cache(&cache);
164
165    world.insert_resource(cache);
166    world.insert_resource(helper);
167
168    world.schedule_scope(lsp_core::Tasks, |_, schedule| {
169        schedule.add_systems(handle_tasks);
170    });
171
172    f(&mut world);
173
174    let (publisher, rx) = DiagnosticPublisher::new();
175    world.insert_resource(publisher);
176
177    world.run_schedule(Startup);
178
179    (world, rx)
180}

182pub fn create_file(
183    world: &mut World,
184    content: &str,
185    url: &str,
186    lang: &str,
187    bundle: impl Bundle,
188) -> Entity {
189    let url = Url::from_str(url).unwrap();
190    let item = TextDocumentItem {
191        version: 1,
192        uri: url.clone(),
193        language_id: String::from(lang),
194        text: String::new(),
195    };
196
197    spawn_or_insert(
198        url.clone(),
199        (
200            Source(content.to_string()),
201            RopeC(ropey::Rope::from_str(content)),
202            Label(url), // this might crash
203            Wrapped(item),
204            Types(HashMap::new()),
205        ),
206        Some(lang.into()),
207        bundle,
208    )(world)
209}

211pub fn debug_world(world: &mut World) {
212    for e in world.query::<Entity>().iter(&world) {
213        let e = world.entity(e);

215            println!("-- Entity {} -- ", l.as_str());
216        } else {
217            println!("-- Nameless entity --");
218        }
219        for c in world.components().iter() {
220            if e.contains_id(c.id()) {
221                println!("c {}", c.name(),);
222            }
223        }
224    }
