13#[instrument(skip(query, commands))]
14pub fn parse_source(
15    query: Query<(Entity, &Source), (Changed<Source>, With<JsonLd>)>,
16    mut commands: Commands,
17) {
18    for (entity, source) in &query {
19        let (tok, es) = tokenize(source.0.as_str());
20        info!("tokenized  {} tokens ({} errors)", tok.len(), es.len());
21        commands.entity(entity).insert((Tokens(tok), Errors(es)));
22    }
23}

25#[instrument(skip(query, commands))]
26pub fn parse_jsonld_system(
27    query: Query<(Entity, &Source, &Tokens, &Label), (Changed<Tokens>, With<JsonLd>)>,
28    mut commands: Commands,
29) {
30    for (entity, source, tokens, label) in &query {
31        let (jsonld, es) = parse(source.as_str(), tokens.0.clone());
32        info!("{} triples ({} errors)", label.0, es.len());
33        if es.is_empty() {
34            let element = Element::<JsonLd>(jsonld);
35            commands
36                .entity(entity)
37                .insert((element, Errors(es)))
38                .remove::<Dirty>();
39        } else {
40            let element = Element::<JsonLd>(jsonld);
41            commands.entity(entity).insert((Errors(es), element, Dirty));
42        }
43    }
44}

46#[instrument(skip(query, commands))]
47pub fn derive_triples(
48    query: Query<(Entity, &Label, &Element<JsonLd>), Changed<Element<JsonLd>>>,
49    mut commands: Commands,
50) {
51    for (e, l, el) in &query {
52        let prefix = triples::derive_prefixes(&el, &l.0);
53        let triples = triples::derive_triples(&el, &prefix);
54        commands.entity(e).insert((Triples(triples), prefix));
55    }
56}
