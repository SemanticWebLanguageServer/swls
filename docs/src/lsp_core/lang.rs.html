19pub trait TokenTrait: Sized {

21
22    fn span_tokens(Spanned(this, span): &Spanned<Self>) -> Vec<(SemanticTokenType, Range<usize>)> {
23        if let Some(x) = this.token() {
24            vec![(x, span.clone())]
25        } else {
26            Vec::new()
27        }
28    }
