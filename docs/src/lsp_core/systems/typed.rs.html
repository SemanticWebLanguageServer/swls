15#[tracing::instrument(skip(query, hierarchy))]
16pub fn extract_type_hierarchy(
17    query: Query<&Triples, (Changed<Triples>, Without<Dirty>)>,
18    mut hierarchy: ResMut<TypeHierarchy<'static>>,
19) {
20    for triples in &query {
21        for q in triples
22            .quads_matching(Any, [rdf::type_], [rdfs::Class], Any)
23            .flatten()
24        {
25            let id = hierarchy.get_id(q.s().as_str());
26
27            for sub_class_of in triples
28                .quads_matching([q.s()], [rdfs::subClassOf], Any, Any)
29                .flatten()
30            {
31                let object = hierarchy.get_id(sub_class_of.o().as_str());
32                hierarchy.set_subclass_of(id, object);
33            }
34
35            for is_sub_class_of in triples
36                .quads_matching(Any, [rdfs::subClassOf], [q.s()], Any)
37                .flatten()
38            {
39                let subject = hierarchy.get_id(is_sub_class_of.o().as_str());
40                hierarchy.set_subclass_of(subject, id);
41            }
42        }
43    }
44}

46pub fn infer_types(
47    mut query: Query<(&Triples, &mut Types), Changed<Triples>>,
48    hierarchy: Res<TypeHierarchy<'static>>,
49) {
50    for (triples, mut types) in &mut query {
51        types.clear();
52
53        for t in triples
54            .quads_matching(Any, [rdf::type_], Any, Any)
55            .flatten()
56        {
57            if let Some(id) = hierarchy.get_id_ref(t.o().as_str()) {
58                let vec = types.0.entry(t.s().value.clone()).or_default();
59                vec.push(id);
60            }
61        }
62    }
63}

65pub fn infer_current_type(
66    query: Query<(Entity, &TripleComponent, &Types)>,
67    mut commands: Commands,
68) {
69    for (e, tc, types) in &query {
70        commands.entity(e).remove::<CurrentType>();
71        if let Some(types) = types.get(tc.triple.s().as_str()) {
72            tracing::debug!(
73                "Found current type for {} {} {:?}",
74                e,
75                tc.triple.s().as_str(),
76                types
77            );
78            commands.entity(e).insert(CurrentType(types.clone()));
79        } else {
80            tracing::debug!("No type found for {}", tc.triple.s().as_str(),);
81        }
82    }
83}

85#[tracing::instrument(skip(query, hierarchy))]
86pub fn hover_types(
87    mut query: Query<(&TokenComponent, &Types, &Prefixes, &mut HoverRequest)>,
88    hierarchy: Res<TypeHierarchy<'static>>,
89) {
90    for (token, types, pref, mut hover) in &mut query {
91        let Some(expaned) = pref.expand(&token.token) else {
92            continue;
93        };
94
95        let Some(types) = types.get(expaned.as_str()) else {
96            continue;
97        };
98
99        for id in types {
100            let type_name = hierarchy.type_name(*id);
101            let type_name = pref
102                .shorten(&type_name)
103                .map(Cow::Owned)
104                .unwrap_or(type_name.clone());
105            hover.0.push(format!("Type: {}", type_name));
106
107            let mut subclass_str = String::from("Sub classes: ");
108            let mut subclasses = hierarchy
109                .iter_subclass(*id)
110                .map(|sub| pref.shorten(&sub).map(Cow::Owned).unwrap_or(sub.clone()))
111                .skip(1);
112
113            if let Some(first) = subclasses.next() {
114                subclass_str += &first;
115                for sub in subclasses {
116                    subclass_str += ", ";
117                    subclass_str += &sub;
118                }
119                hover.0.push(subclass_str);
120            }
121
122            let mut subclass_str = String::from("Super classes: ");
123            let mut subclasses = hierarchy
124                .iter_superclass(*id)
125                .map(|sub| pref.shorten(&sub).map(Cow::Owned).unwrap_or(sub.clone()))
126                .skip(1);
127
128            if let Some(first) = subclasses.next() {
129                subclass_str += &first;
130                for sub in subclasses {
131                    subclass_str += ", ";
132                    subclass_str += &sub;
133                }
134                hover.0.push(subclass_str);
135            }
136        }
137    }
