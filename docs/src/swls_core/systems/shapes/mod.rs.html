27pub fn derive_shapes(
28    query: Query<(Entity, &Label, &Triples), (Changed<Triples>, Without<Dirty>)>,
29    mut commands: Commands,
30    res: Res<ServerConfig>,
31) {
32    if res.config.local.disabled.contains(&Disabled::Shapes) {
33        return;
34    }
35    for (e, label, data) in &query {
36        let _entered = info_span!("derive_shape", label = label.as_str()).entered();
37        commands.entity(e).remove::<ShaclShapes>();
38
39        let rdf = Rdf::new(data);
40
41        let Some(schema) = shacl_schema_from_data(rdf) else {
42            continue;
43        };
44
45        let Some(shacl_ir) = ShaclSchemaIR::compile(&schema).ok() else {
46            continue;
47        };
48        info!("Found shacl ir with {} shapes", shacl_ir.iter().count());
49
50        commands.entity(e).insert(ShaclShapes::new(shacl_ir));
51    }
52}

54/// System evaluates linked shapes
55#[instrument(skip(query, other, client, res))]
56pub fn validate_shapes(
57    query: Query<
58        (
59            &RopeC,
60            &Label,
61            &DocumentLinks,
62            &Wrapped<TextDocumentItem>,
63            &Prefixes,
64            &Triples,
65        ),
66        (Changed<Triples>, With<Open>),
67    >,
68    other: Query<(&Label, &ShaclShapes, Option<&Global>, &Triples)>,
69    mut client: ResMut<DiagnosticPublisher>,
70    res: Res<ServerConfig>,
71) {
72    if res.config.local.disabled.contains(&Disabled::Shapes) {
73        return;
74    }
75
76    for (rope, label, links, item, prefixes, triples) in &query {
77        info!("Validate shapes {}", label.as_str());
78        let client: &mut DiagnosticPublisher = &mut client;
79        let mut diagnostics: Vec<crate::lsp_types::Diagnostic> = Vec::new();
80
81        let rdf = Rdf::new(triples);
82        let mut validation_results = Vec::new();
83        let mut runner = NativeEngine::new();
84
85        for (other_label, schema, global, shape_triples) in &other {
86            if global.is_none()
87                && links
88                    .iter()
89                    .find(|link| link.0.as_str().starts_with(other_label.0.as_str()))
90                    .is_none()
91                && label.0 != other_label.0
92            {
93                continue;
94            }
95
96            let _entered = info_span!(
97                "validate_shapes",
98                data = label.as_str(),
99                shapes = other_label.as_str()
100            )
101            .entered();
102
103            let ir = schema.ir();
104            for (_, shape) in ir.iter_with_targets() {
105                let results = match shape.validate(&rdf, &mut runner, None, Some(shape), ir) {
106                    Ok(r) => r,
107                    Err(e) => {
108                        tracing::debug!("errors {}", e);
109                        continue;
110                    }
111                };
112
113                let mut map = HashMap::new();
114
115                for res in &results {
116                    let key = (res.source(), res.path());
117                    let entry: &mut Vec<&ValidationResult> = map.entry(key).or_default();
118                    entry.push(res);
119                }
120
121                for (_, all) in map {
122                    let res = all[0];
123                    let severity = match res.severity() {
124                        shacl_ir::severity::CompiledSeverity::Info => {
125                            Some(crate::lsp_types::DiagnosticSeverity::INFORMATION)
126                        }
127                        shacl_ir::severity::CompiledSeverity::Warning => {
128                            Some(crate::lsp_types::DiagnosticSeverity::WARNING)
129                        }
130                        shacl_ir::severity::CompiledSeverity::Violation => {
131                            Some(crate::lsp_types::DiagnosticSeverity::ERROR)
132                        }
133                        _ => None,
134                    };
135
136                    let term: MyTerm = MyTerm::from(res.focus_node().clone());
137                    let related_subjects = triples.iter().map(|x| x.s()).filter(|&x| x == &term);
138                    let related_objects = triples.iter().map(|x| x.o()).filter(|&x| x == &term);
139                    let Some(span) = related_subjects
140                        .chain(related_objects)
141                        .min_by_key(|x| x.span.start)
142                        .map(|x| x.span.clone())
143                    else {
144                        continue;
145                    };
146                    let Some(range) = util::range_to_range(&span, rope) else {
147                        continue;
148                    };
149
150                    let mut writer = String::new();
151
152                    for res in all {
153                        if let Some(m) = res.message() {
154                            write!(&mut writer, "{}\n", m).unwrap();
155                        } else {
156                            write!(&mut writer, "Component {} ", res.component().to_string())
157                                .unwrap();
158                        }
159                    }
160
161                    if let Some(path) = res.path() {
162                        write!(
163                            &mut writer,
164                            "on path {} ",
165                            PrefixedPath::new(path, prefixes)
166                        )
167                        .unwrap();
168                    }
169
170                    let id = MyTerm::from(shape.id().clone());
171                    if id.ty == Some(sophia_api::term::TermKind::Iri) {
172                        write!(&mut writer, "({})\n", id.value).unwrap();
173                    } else {
174                        write!(&mut writer, "\n",).unwrap();
175                    }
176
177                    if let Some(source) = res.source().cloned() {
178                        let source = MyTerm::from(source);
179
180                        if let Some(message) = shape_triples.object(
181                            [&source],
182                            [
183                                &MyTerm::named_node(
184                                    "http://www.w3.org/2000/01/rdf-schema#comment",
185                                    0..0,
186                                ),
187                                &MyTerm::named_node("http://www.w3.org/ns/shacl#message", 0..0),
188                            ],
189                        ) {
190                            write!(
191                                &mut writer,
192                                "\n{}\n",
193                                message
194                                    .value
195                                    .split_whitespace()
196                                    .collect::<Vec<_>>()
197                                    .join(" ")
198                            )
199                            .unwrap();
200                        }
201                    }
202
203                    diagnostics.push(crate::lsp_types::Diagnostic {
204                        range,
205                        severity,
206                        source: Some(String::from("SWLS")),
207                        message: writer,
208                        ..Default::default()
209                    });
210                }
211
212                validation_results.extend(results);
213            }
214        }
215
216        let _ = client.publish(&item.0, diagnostics, "shacl_validation");
217    }
