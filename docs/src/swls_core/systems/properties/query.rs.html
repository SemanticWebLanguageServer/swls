9pub fn find_classes(
10    store: &oxigraph::store::Store,
11) -> Option<HashMap<oxigraph::model::NamedNode, DefinedClass>> {
12    use oxigraph::model::{Literal, NamedNode};
13    use oxigraph::sparql::{QueryResults, SparqlEvaluator};
14
15    let mut definitions = HashMap::new();
16    if let QueryResults::Solutions(solutions) = SparqlEvaluator::new()
17        .parse_query(include_str!("../../queries/class_definitions.sparql"))
18        .ok()?
19        .on_store(&store)
20        .execute()
21        .ok()?
22    {
23        for solution in solutions {
24            if let Ok(solution) = solution {
25                let class = solution.get("class").unwrap();
26                let title = solution.get("title");
27                let description = solution.get("description");
28
29                match (
30                    NamedNode::try_from(class.clone()),
31                ) {
32                    (
33                        Ok(class),
34                    ) => {
35                        let v = definitions
36                            .entry(class.clone())
37                            .or_insert_with(|| DefinedClass {
38                                term: MyTerm::named_node(class.as_str(), 0..0).to_owned(),
39                                locations: std::collections::HashSet::new(),
40                                titles: std::collections::HashSet::new(),
41                                descriptions: std::collections::HashSet::new(),
42                            });
43
44                        if let Some(title) = title.and_then(|x| Literal::try_from(x.clone()).ok()) {
45                            v.titles.insert(title.value().to_string());
46                        }
47
48                        if let Some(description) =
49                            description.and_then(|x| Literal::try_from(x.clone()).ok())
50                        {
51                            v.descriptions.insert(description.value().to_string());
52                        }
53                    }
54                    _ => {}
55                }
56            }
57        }
58    }
59
60    Some(definitions)
61}

63pub fn find_properties(store: &oxigraph::store::Store) -> Option<HashMap<String, DefinedProperty>> {
64    use oxigraph::model::{Literal, NamedNode};
65    use oxigraph::sparql::{QueryResults, SparqlEvaluator};
66
67    let mut definitions = HashMap::new();
68    if let QueryResults::Solutions(solutions) = SparqlEvaluator::new()
69        .parse_query(include_str!("../../queries/property_definitions.sparql"))
70        .ok()?
71        .on_store(&store)
72        .execute()
73        .ok()?
74    {
75        for solution in solutions {
76            if let Ok(solution) = solution {
77                let class = solution.get("property").unwrap();
78                let title = solution.get("title");
79                let description = solution.get("description");
80
81                let range = solution
82                    .get("range")
83                    .and_then(|r| NamedNode::try_from(r.clone()).ok())
84                    .map(|r| MyTerm::named_node(r.as_str(), 0..0).to_owned());
85                let domain = solution
86                    .get("domain")
87                    .and_then(|r| NamedNode::try_from(r.clone()).ok())
88                    .map(|r| MyTerm::named_node(r.as_str(), 0..0).to_owned());
89
90                match (
91                    NamedNode::try_from(class.clone()),
92                ) {
93                    (
94                        Ok(class),
95                    ) => {
96                        let v = definitions
97                            .entry(class.as_str().to_string())
98                            .or_insert_with(|| DefinedProperty {
99                                term: MyTerm::named_node(class.as_str(), 0..0).to_owned(),
100                                locations: std::collections::HashSet::new(),
101                                titles: std::collections::HashSet::new(),
102                                descriptions: std::collections::HashSet::new(),
103                                ranges: std::collections::HashSet::new(),
104                                domains: std::collections::HashSet::new(),
105                            });
106
107                        if let Some(title) = title.and_then(|x| Literal::try_from(x.clone()).ok()) {
108                            v.titles.insert(title.value().to_string());
109                        }
110
111                        if let Some(description) =
112                            description.and_then(|x| Literal::try_from(x.clone()).ok())
113                        {
114                            v.descriptions.insert(description.value().to_string());
115                        }
116                        if let Some(domain) = domain {
117                            v.domains.insert(domain);
118                        }
119
120                        if let Some(range) = range {
121                            v.ranges.insert(range);
122                        }
123                    }
124                    _ => {}
125                }
126            }
127        }
128    }
129
130    Some(definitions)
131}

133pub fn derive_ontologies(
134    query: Query<(), Added<Triples>>,
135    store: Res<Store>,
136    mut resource: ResMut<Ontologies>,
137    mut hierarchy: ResMut<TypeHierarchy<'static>>,
138) {
139    if !query.is_empty() {
140        if let Some(classes) = find_classes(&store.0) {
141            resource.classes = classes;
142        }
143
144        if let Some(properties) = find_properties(&store.0) {
145            for p in properties.values() {
146                for domain in &p.domains {
147                    let _ = hierarchy.get_id(domain.as_str());
148                }
149
150                for range in &p.ranges {
151                    let _ = hierarchy.get_id(range.as_str());
152                }
153            }
154
155            resource.properties = properties;
156        }
157    }
158}
