13fn sparql_kwd(
14    kwd: SparqlKeyword,
15) -> impl Parser<Token, Spanned<SparqlKeyword>, Error = Simple<Token>> + Clone {
16    just(Token::SparqlKeyword(kwd.clone()))
17        .to(kwd)
18        .map_with_span(spanned)

21fn prologue() -> impl Parser<Token, Prologue, Error = Simple<Token>> + Clone {
22    let base = named_node()
23        .map_with_span(spanned)
24        .then(just(Token::SparqlBase).map_with_span(spanned))
25        .map(|(iri, token)| Prologue::Base { token, iri });
26
27    let prefix = named_node()
28        .map_with_span(spanned)
29        .then(select! { |span| Token::PNameLN(x, _) => Spanned(x.unwrap_or_default(), span)})
30        .then(just(Token::SparqlPrefix).map_with_span(|_, s| s))
31        .map(|((value, prefix), span)| {
32            Prologue::Prefix(TurtlePrefix {
33                span,
34                prefix,
35                value,
36            })
37        });
38
39    base.or(prefix)
40}

42fn dataset_clause() -> impl Parser<Token, DatasetClause, Error = Simple<Token>> + Clone {
43    named_node()
44        .map_with_span(spanned)
45        .then(sparql_kwd(SparqlKeyword::Named).or_not())
46        .then(sparql_kwd(SparqlKeyword::From))
47        .map(|((iri, named), from)| DatasetClause { from, named, iri })
48}

68fn select_clause() -> impl Parser<Token, SelectClause, Error = Simple<Token>> + Clone {
69    let star = just(Token::SparqlExpr(SparqlExpr::Times))
70        .to(Solution::All)
71        .map_with_span(spanned)
72        .map(|x| vec![x]);
73
74    let others = bind()
75        .delimited_by(just(Token::CurlClose), just(Token::CurlOpen))
76        .map(Solution::VarAs)
77        .or(variable().map(Solution::Var))
78        .map_with_span(spanned)
79        .repeated();
80
81    // This might be different to support ask and describe
82    star.or(others)
83        .then(
84            sparql_kwd(SparqlKeyword::Distinct)
85                .or(sparql_kwd(SparqlKeyword::Reduced))
86                .or_not(),
87        )
88        .then(sparql_kwd(SparqlKeyword::Select))
89        .map(|((solutions, modifier), kwd)| SelectClause {
90            kwd,
91            modifier,
92            solutions,
93        })
94}

96fn sub_select() -> impl Parser<Token, SubSelect, Error = Simple<Token>> + Clone {
97    recursive(|sub_select| {
98        let modi = modifier().map_with_span(spanned).repeated();
99        modi.then(where_clause(sub_select))
100            .then(select_clause())
101            .map(|((modifier, where_clause), select)| SubSelect {
102                modifier,
103                where_clause,
104                select,
105            })
106    })
107}

109fn group_graph_pattern_sub(
110    ggp: impl Parser<Token, GroupGraphPattern, Error = Simple<Token>> + Clone,
111) -> impl Parser<Token, GroupGraphPatternSub, Error = Simple<Token>> + Clone {
112    let next_check = none_of([Token::CurlOpen]).rewind();
113
114    let trip = triple()
115        .map_with_span(spanned)
116        .map(GroupGraphPatternSub::Triple)
117        .labelled("triple");
118    let kwd = just(Token::CurlClose)
119        .rewind()
120        .ignore_then(ggp.clone())
121        .clone()
122        .map_with_span(spanned)
123        .then(sparql_kwd(SparqlKeyword::Minus).or(sparql_kwd(SparqlKeyword::Optional)))
124        .map(|(ggp, kwd)| GroupGraphPatternSub::Kwd(kwd, ggp));
125
126    let union = just(Token::CurlClose)
127        .rewind()
128        .ignore_then(ggp.clone())
129        .map_with_span(spanned)
130        .then(
131            sparql_kwd(SparqlKeyword::Union)
132                .then(ggp.map_with_span(spanned))
133                .repeated(),
134        )
135        .map(|(start, rest)| GroupGraphPatternSub::Union(start, rest));
136
137    // TODO add the others
138    //
139
140    next_check.ignore_then(trip.or(kwd).or(union).labelled("group_graph_pattern_sub"))
141}

143fn expect_it(
144    token: Token,
145    st: &'static str,
146) -> impl Parser<Token, Token, Error = Simple<Token>> + Clone {
147    just(token.clone()).or(none_of([token.clone()]).try_map(move |x: Token, span| {
148        println!("{} didn't expect {}", st, x);
149        Err(Simple::expected_input_found(
150            span,
151            [Some(token.clone())],
152            Some(x.clone()),
153        ))

157fn group_graph_pattern(
158    select: impl Parser<Token, SubSelect, Error = Simple<Token>> + Clone + 'static,
159) -> impl Parser<Token, GroupGraphPattern, Error = Simple<Token>> + Clone {
160    let s = select.clone();
161    recursive(|ggp| {
162        let select = s
163            .clone()
164            .map(Box::from)
165            .map(GroupGraphPattern::SubSelect)
166            .labelled("sub_select");
167
168        let gg = group_graph_pattern_sub(ggp)
169            .map_with_span(spanned)
170            .repeated()
171            .map(GroupGraphPattern::GroupGraph);
172
173        let close = expect_it(Token::CurlClose, "close").labelled("CurlClose");
174        let open = expect_it(Token::CurlOpen, "open").labelled("CurlOpen");
175
176        close.ignore_then(gg.or(select)).then_ignore(open)
177    })
178}

180fn where_clause(
181    select: impl Parser<Token, SubSelect, Error = Simple<Token>> + Clone + 'static,
182) -> impl Parser<Token, WhereClause, Error = Simple<Token>> + Clone {
183    group_graph_pattern(select)
184        .map_with_span(spanned)
185        .then(sparql_kwd(SparqlKeyword::Where).or_not())
186        .map(|(ggp, kwd)| WhereClause { ggp, kwd })
187}

189fn modifier() -> impl Parser<Token, Modifier, Error = Simple<Token>> + Clone {
190    let num = select!(
191        Token::Number(x) => x,
192    )
193    .map_with_span(spanned);
194    let limit_offset = num
195        .then(sparql_kwd(SparqlKeyword::Limit).or(sparql_kwd(SparqlKeyword::Offset)))
196        .map(|(num, kwd)| Modifier::LimitOffset(kwd, num));
197    limit_offset
198}

200pub fn query(base: lsp_types::Url) -> impl Parser<Token, Query, Error = Simple<Token>> + Clone {
201    let prologues = prologue().map_with_span(spanned).repeated().map(|xs| {
202        let mut base = None;
203        let mut prefixes = vec![];
204        xs.into_iter().for_each(|Spanned(x, span)| match x {
205            Prologue::Base { token, iri } => base = Some(Spanned(Base { token, iri }, span)),
206            Prologue::Prefix(prefix) => prefixes.push(Spanned(prefix, span)),
207        });
208        (base, prefixes)
209    });
210    let kwds = select_clause().map(QueryClause::Select);
211    let datasets = dataset_clause().map_with_span(spanned).repeated();
212    let where_clause = where_clause(sub_select()).map_with_span(spanned);
213    let modifiers = modifier().map_with_span(spanned).repeated();
214
215    modifiers
216        .then(where_clause)
217        .then(datasets)
218        .then(kwds)
219        .then(prologues)
220        .map(
221            move |((((modifier, where_clause), datasets), kwds), (base_statement, prefixes))| {
222                Query {
223                    base_statement,
224                    prefixes,
225                    base: base.clone(),
226                    modifier,
227                    where_clause,
228                    datasets,
229                    kwds,
230                }
231            },
232        )
233}

235pub fn parse(
236    source: &str,
237    base: lsp_types::Url,
238    tokens: Vec<Spanned<Token>>,
239) -> (Spanned<Query>, Vec<(usize, Simple<Token>)>) {
240    let len = source.len();
241    let rev_range = |range: std::ops::Range<usize>| (len - range.end)..(len - range.start);
242    let stream = chumsky::Stream::from_iter(
243        0..len,
244        tokens
245            .into_iter()
246            .rev()
247            .filter(|x| !x.is_comment())
248            .map(|Spanned(x, s)| (x, rev_range(s))),
249    );
250
251    let parser = query(base)
252        .map_with_span(spanned)
253        .then_ignore(end().recover_with(skip_then_retry_until([])));
254    let (mut json, json_errors) = parser.parse_recovery(stream);
255
256    json.iter_mut().for_each(|query| query.fix_spans(len));
257    let json_errors: Vec<_> = json_errors.into_iter().map(|error| (len, error)).collect();
258    (
259        json.unwrap_or(Spanned(Query::default(), 0..source.len())),
260        json_errors,
261    )
262}

276            println!("token {:?}", token);

278        let end = turtle.len()..turtle.len();
279        let stream = Stream::from_iter(
280            end,
281            tokens
282                .into_iter()
283                .map(|Spanned(x, y)| (x, y))
284                .rev()
285                .filter(|x| !x.0.is_comment()),
286        );
287
288        parser
289            .then_ignore(chumsky::prelude::end())
290            .parse_recovery(stream)
291    }
