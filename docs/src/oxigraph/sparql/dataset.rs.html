19pub struct DatasetView<'a> {
20    reader: StorageReader<'a>,
21    extra: RefCell<HashMap<StrHash, String, BuildHasherDefault<StrHashHasher>>>,
22}
23
24impl<'a> DatasetView<'a> {
25    pub fn new(reader: StorageReader<'a>) -> Self {
26        Self {
27            reader,
28            extra: RefCell::new(HashMap::default()),
29        }
30    }
31
32    pub fn insert_str(&self, key: &StrHash, value: &str) {
33        if let Entry::Vacant(e) = self.extra.borrow_mut().entry(*key) {
34            if !matches!(self.reader.contains_str(key), Ok(true)) {
35                e.insert(value.to_owned());
36            }
37        }
38    }
39}

41impl<'a> QueryableDataset<'a> for DatasetView<'a> {
42    type InternalTerm = EncodedTerm;
43    type Error = StorageError;
44
45    fn internal_quads_for_pattern(
46        &self,
47        subject: Option<&EncodedTerm>,
48        predicate: Option<&EncodedTerm>,
49        object: Option<&EncodedTerm>,
50        graph_name: Option<Option<&EncodedTerm>>,
51    ) -> impl Iterator<Item = Result<InternalQuad<EncodedTerm>, StorageError>> + use<'a> {
52        self.reader
53            .quads_for_pattern(
54                subject,
55                predicate,
56                object,
57                graph_name.map(|graph_name| graph_name.unwrap_or(&EncodedTerm::DefaultGraph)),
58            )
59            .map(|quad| {
60                let quad = quad?;
61                Ok(InternalQuad {
62                    subject: quad.subject,
63                    predicate: quad.predicate,
64                    object: quad.object,
65                    graph_name: if quad.graph_name.is_default_graph() {
66                        None
67                    } else {
68                        Some(quad.graph_name)
69                    },
70                })
71            })
72    }
73
74    fn internal_named_graphs(
75        &self,
76    ) -> impl Iterator<Item = Result<EncodedTerm, StorageError>> + use<'a> {
77        self.reader.named_graphs()
78    }
79
80    fn contains_internal_graph_name(&self, graph_name: &EncodedTerm) -> Result<bool, StorageError> {
81        self.reader.contains_named_graph(graph_name)
82    }
83
84    fn internalize_term(&self, term: Term) -> Result<EncodedTerm, StorageError> {
85        let encoded = term.as_ref().into();
86        insert_term(term.as_ref(), &encoded, &mut |key, value| {
87            self.insert_str(key, value)
88        });
89        Ok(encoded)
90    }
91
92    fn externalize_term(&self, term: EncodedTerm) -> Result<Term, StorageError> {
93        self.decode_term(&term)
94    }
95
96    fn externalize_expression_term(
97        &self,
98        term: EncodedTerm,
99    ) -> Result<ExpressionTerm, StorageError> {
100        Ok(match term {
101            EncodedTerm::DefaultGraph => {
102                return Err(CorruptionError::new("Unexpected default graph").into());
103            }
104            EncodedTerm::BooleanLiteral(value) => ExpressionTerm::BooleanLiteral(value),
105            EncodedTerm::FloatLiteral(value) => ExpressionTerm::FloatLiteral(value),
106            EncodedTerm::DoubleLiteral(value) => ExpressionTerm::DoubleLiteral(value),
107            EncodedTerm::IntegerLiteral(value) => ExpressionTerm::IntegerLiteral(value),
108            EncodedTerm::DecimalLiteral(value) => ExpressionTerm::DecimalLiteral(value),
109            EncodedTerm::DateTimeLiteral(value) => ExpressionTerm::DateTimeLiteral(value),
110            EncodedTerm::TimeLiteral(value) => ExpressionTerm::TimeLiteral(value),
111            EncodedTerm::DateLiteral(value) => ExpressionTerm::DateLiteral(value),
112            EncodedTerm::GYearMonthLiteral(value) => ExpressionTerm::GYearMonthLiteral(value),
113            EncodedTerm::GYearLiteral(value) => ExpressionTerm::GYearLiteral(value),
114            EncodedTerm::GMonthDayLiteral(value) => ExpressionTerm::GMonthDayLiteral(value),
115            EncodedTerm::GDayLiteral(value) => ExpressionTerm::GDayLiteral(value),
116            EncodedTerm::GMonthLiteral(value) => ExpressionTerm::GMonthLiteral(value),
117            EncodedTerm::DurationLiteral(value) => ExpressionTerm::DurationLiteral(value),
118            EncodedTerm::YearMonthDurationLiteral(value) => {
119                ExpressionTerm::YearMonthDurationLiteral(value)
120            }
121            EncodedTerm::DayTimeDurationLiteral(value) => {
122                ExpressionTerm::DayTimeDurationLiteral(value)
123            }
124            #[cfg(feature = "rdf-12")]
125            EncodedTerm::Triple(t) => ExpressionTriple::new(
126                self.externalize_expression_term(t.subject.clone())?,
127                self.externalize_expression_term(t.predicate.clone())?,
128                self.externalize_expression_term(t.object.clone())?,
129            )
130            .ok_or_else(|| CorruptionError::msg("Invalid triple term in the storage"))?
131            .into(),
132            _ => self.decode_term(&term)?.into(), // No escape
133        })
134    }
135
136    fn internalize_expression_term(
137        &self,
138        term: ExpressionTerm,
139    ) -> Result<EncodedTerm, StorageError> {
140        Ok(match term {
141            ExpressionTerm::BooleanLiteral(value) => EncodedTerm::BooleanLiteral(value),
142            ExpressionTerm::FloatLiteral(value) => EncodedTerm::FloatLiteral(value),
143            ExpressionTerm::DoubleLiteral(value) => EncodedTerm::DoubleLiteral(value),
144            ExpressionTerm::IntegerLiteral(value) => EncodedTerm::IntegerLiteral(value),
145            ExpressionTerm::DecimalLiteral(value) => EncodedTerm::DecimalLiteral(value),
146            ExpressionTerm::DateTimeLiteral(value) => EncodedTerm::DateTimeLiteral(value),
147            ExpressionTerm::TimeLiteral(value) => EncodedTerm::TimeLiteral(value),
148            ExpressionTerm::DateLiteral(value) => EncodedTerm::DateLiteral(value),
149            ExpressionTerm::GYearMonthLiteral(value) => EncodedTerm::GYearMonthLiteral(value),
150            ExpressionTerm::GYearLiteral(value) => EncodedTerm::GYearLiteral(value),
151            ExpressionTerm::GMonthDayLiteral(value) => EncodedTerm::GMonthDayLiteral(value),
152            ExpressionTerm::GDayLiteral(value) => EncodedTerm::GDayLiteral(value),
153            ExpressionTerm::GMonthLiteral(value) => EncodedTerm::GMonthLiteral(value),
154            ExpressionTerm::DurationLiteral(value) => EncodedTerm::DurationLiteral(value),
155            ExpressionTerm::YearMonthDurationLiteral(value) => {
156                EncodedTerm::YearMonthDurationLiteral(value)
157            }
158            ExpressionTerm::DayTimeDurationLiteral(value) => {
159                EncodedTerm::DayTimeDurationLiteral(value)
160            }
161            #[cfg(feature = "rdf-12")]
162            ExpressionTerm::Triple(t) => EncodedTerm::Triple(Arc::new(EncodedTriple {
163                subject: self.internalize_expression_term(t.subject.into())?,
164                predicate: self.internalize_expression_term(t.predicate.into())?,
165                object: self.internalize_expression_term(t.object)?,
166            })),
167            _ => self.internalize_term(term.into())?, // No fast path
168        })
169    }
170
171    fn internal_term_effective_boolean_value(
172        &self,
173        term: EncodedTerm,
174    ) -> Result<Option<bool>, StorageError> {
175        Ok(match term {
176            EncodedTerm::BooleanLiteral(value) => Some(value.into()),
177            EncodedTerm::SmallStringLiteral(value) => Some(!value.is_empty()),
178            EncodedTerm::BigStringLiteral { .. } => {
179                Some(false) // A big literal can't be empty
180            }
181            EncodedTerm::FloatLiteral(value) => Some(Boolean::from(value).into()),
182            EncodedTerm::DoubleLiteral(value) => Some(Boolean::from(value).into()),
183            EncodedTerm::IntegerLiteral(value) => Some(Boolean::from(value).into()),
184            EncodedTerm::DecimalLiteral(value) => Some(Boolean::from(value).into()),
185            _ => None,
186        })
187    }
188}

190impl StrLookup for DatasetView<'_> {
191    fn get_str(&self, key: &StrHash) -> Result<Option<String>, StorageError> {
192        Ok(if let Some(value) = self.extra.borrow().get(key) {
193            Some(value.clone())
194        } else {
195            self.reader.get_str(key)?
196        })
197    }
