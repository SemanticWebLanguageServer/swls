13pub struct Client {
14    client: Arc<oxhttp::Client>,
15}
16
17impl Client {
18    pub fn new(timeout: Option<Duration>, redirection_limit: usize) -> Self {
19        let mut client = oxhttp::Client::new()
20            .with_redirection_limit(redirection_limit)
21            .with_user_agent(concat!("Oxigraph/", env!("CARGO_PKG_VERSION")))
22            .unwrap();
23        if let Some(timeout) = timeout {
24            client = client.with_global_timeout(timeout);
25        }
26        Self {
27            client: Arc::new(client),
28        }
29    }
30
31    pub fn get(&self, url: &str, accept: &'static str) -> Result<(String, impl Read)> {
32        let request = Request::builder()
33            .uri(url)
34            .header(ACCEPT, accept)
35            .body(())
36            .map_err(invalid_input_error)?;
37        let response = self.client.request(request)?;
38        let status = response.status();
39        if !status.is_success() {
40            return Err(Error::other(format!(
41                "Error {} returned by {} with payload:\n{}",
42                status,
43                url,
44                response.into_body().to_string()?
45            )));
46        }
47        let content_type = response
48            .headers()
49            .get(CONTENT_TYPE)
50            .ok_or_else(|| invalid_data_error(format!("No Content-Type returned by {url}")))?
51            .to_str()
52            .map_err(invalid_data_error)?
53            .to_owned();
54        Ok((content_type, response.into_body()))
55    }
56
57    pub fn post(
58        &self,
59        url: &str,
60        payload: Vec<u8>,
61        content_type: &'static str,
62        accept: &'static str,
63    ) -> Result<(String, Body)> {
64        let request = Request::builder()
65            .method(Method::POST)
66            .uri(url)
67            .header(ACCEPT, accept)
68            .header(CONTENT_TYPE, content_type)
69            .body(payload)
70            .map_err(invalid_input_error)?;
71        let response = self.client.request(request)?;
72        let status = response.status();
73        if !status.is_success() {
74            return Err(Error::other(format!(
75                "Error {} returned by {} with payload:\n{}",
76                status,
77                url,
78                response.into_body().to_string()?
79            )));
80        }
81        let content_type = response
82            .headers()
83            .get(CONTENT_TYPE)
84            .ok_or_else(|| invalid_data_error(format!("No Content-Type returned by {url}")))?
85            .to_str()
86            .map_err(invalid_data_error)?
87            .to_owned();
88        Ok((content_type, response.into_body()))
89    }
90}

104impl DefaultServiceHandler for HttpServiceHandler {
105    type Error = QueryEvaluationError;
106
107    fn handle(
108        &self,
109        service_name: &NamedNode,
110        pattern: &GraphPattern,
111        base_iri: Option<&Iri<String>>,
112    ) -> std::result::Result<QuerySolutionIter<'static>, Self::Error> {
113        let (content_type, body) = self
114            .client
115            .post(
116                service_name.as_str(),
117                spargebra::Query::Select {
118                    dataset: None,
119                    pattern: pattern.clone(),
120                    base_iri: base_iri.cloned(),
121                }
122                .to_string()
123                .into_bytes(),
124                "application/sparql-query",
125                "application/sparql-results+json, application/sparql-results+xml",
126            )
127            .map_err(|e| QueryEvaluationError::Service(Box::new(e)))?;
128        let format = QueryResultsFormat::from_media_type(&content_type).ok_or_else(|| {
129            QueryEvaluationError::Service(
130                format!(
131                    "Unsupported Content-Type returned by service {service_name}: {content_type}"
132                )
133                .into(),
134            )
135        })?;
136        let ReaderQueryResultsParserOutput::Solutions(reader) =
137            QueryResultsParser::from_format(format)
138                .for_reader(body)
139                .map_err(|e| QueryEvaluationError::Service(Box::new(e)))?
140        else {
141            return Err(QueryEvaluationError::Service(
142                "No valid SPARQL solutions returned by {service_name}".into(),
143            ));
144        };
145        Ok(QuerySolutionIter::new(
146            reader.variables().into(),
147            Box::new(reader.map(|t| t.map_err(|e| QueryEvaluationError::Service(Box::new(e))))),
148        ))
149    }
150}

152fn invalid_data_error(error: impl Into<Box<dyn std::error::Error + Send + Sync>>) -> Error {
