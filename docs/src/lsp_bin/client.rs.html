30impl ClientSync for TowerClient {
31    fn spawn<F: std::future::Future<Output = ()> + Send + 'static>(&self, fut: F) {
32        let handle = std::thread::current();
33        info!("Spawn threaad name {:?}", handle.id());
34        self.handle.spawn(fut);
35        // info!("Should spawn but won't!");
36    }
37
38    fn fetch(
39        &self,
40        url: &str,
41        headers: &HashMap<String, String>,
42    ) -> Pin<Box<dyn Send + std::future::Future<Output = Result<Resp, String>>>> {
43        use tokio::{fs::File, io::AsyncReadExt};
44        use tracing::{debug, error, info};
45        info!("Should fetch, fetching!");
46
47        let m_url = reqwest::Url::parse(url);
48
49        let client = ::reqwest::Client::new();
50        let builder = client.get(url);
51        let builder = headers
52            .into_iter()
53            .fold(builder, |builder, (k, v)| builder.header(k, v));
54
55        return async {
56            let url = m_url.map_err(|_| String::from("invalid url!"))?;
57            info!("Found url {} {}", url.scheme(), url);
58            if url.scheme() == "file" {
59                let mut file = File::open(url.path())
60                    .await
61                    .map_err(|_| format!("File not found {}", url.path()))?;
62                let mut body = String::new();
63                file.read_to_string(&mut body)
64                    .await
65                    .map_err(|_| format!("Failed to read file"))?;
66                let status = 200;
67                let headers = Vec::new();
68                return Ok(Resp {
69                    headers,
70                    body,
71                    status,
72                });
73            }
74
75            debug!("sending blocking");
76            let resp = match builder.send().await {
77                Ok(x) => x,
78                Err(e) => {
79                    error!(error = ?e);
80                    return Err(e.to_string());
81                }
82            };
83
84            let status = resp.status().as_u16();
85            let headers: Vec<_> = resp
86                .headers()
87                .iter()
88                .flat_map(|(h, v)| {
89                    if let Ok(value) = v.to_str() {
90                        Some((h.to_string(), value.to_string()))
91                    } else {
92                        None
93                    }
94                })
95                .collect();
96            debug!("got resp");
97            let body = resp.text().await.unwrap();
98
99            Ok(Resp {
100                headers,
101                body,
102                status,
103            })
104        }
105        .boxed();
106    }
107}
