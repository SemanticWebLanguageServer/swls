10pub(crate) fn node_kind<RDF: FocusRDF>() -> impl RDFNodeParse<RDF, Output = Vec<Component>> {
11    ValuesPropertyParser::new(ShaclVocab::sh_node_kind().clone()).flat_map(|ns| {
12        let nks: Vec<_> = ns
13            .into_iter()
14            .flat_map(|term| {
15                let nk = term_to_node_kind::<RDF>(term)?;
16                Ok::<Component, ShaclParserError>(Component::NodeKind(nk))
17            })
18            .collect();
19        Ok(nks)
20    })
21}
22
23fn term_to_node_kind<RDF: Rdf>(term: RDF::Term) -> Result<NodeKind, ShaclParserError> {
24    let term_name = term.to_string();
25    let result_iri: Result<RDF::IRI, ShaclParserError> =
26        <RDF::Term as TryInto<RDF::IRI>>::try_into(term).map_err(|_| ShaclParserError::ExpectedNodeKind {
27            term: term_name.to_string(),
28        });
29    let iri = result_iri?;
30    match iri.as_str() {
31        ShaclVocab::SH_IRI => Ok(NodeKind::Iri),
32        ShaclVocab::SH_LITERAL => Ok(NodeKind::Literal),
33        ShaclVocab::SH_BLANK_NODE => Ok(NodeKind::BlankNode),
34        ShaclVocab::SH_BLANK_NODE_OR_IRI => Ok(NodeKind::BlankNodeOrIri),
35        ShaclVocab::SH_BLANK_NODE_OR_LITERAL => Ok(NodeKind::BlankNodeOrLiteral),
36        ShaclVocab::SH_IRI_OR_LITERAL => Ok(NodeKind::IRIOrLiteral),
37        _ => Err(ShaclParserError::UnknownNodeKind { term: term_name }),
38    }
39}
