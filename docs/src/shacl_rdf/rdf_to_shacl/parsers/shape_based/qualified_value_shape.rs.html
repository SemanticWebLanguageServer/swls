13/// This parser gets the siblings of a focus node
14/// Siblings are the other qualified value shapes that share the same parent(s)
15/// The defnition in the spec is: https://www.w3.org/TR/shacl12-core/#dfn-sibling-shapes
16/// "Let Q be a shape in shapes graph G that declares a qualified cardinality constraint
17/// (by having values for sh:qualifiedValueShape and at least one of sh:qualifiedMinCount
18/// or sh:qualifiedMaxCount).
19/// Let ps be the set of shapes in G that have Q as a value of sh:property.
20/// If Q has true as a value for sh:qualifiedValueShapesDisjoint then the set of sibling
21/// shapes for Q is defined as the set of all values of the SPARQL property path
22/// sh:property/sh:qualifiedValueShape for any shape in ps minus the value of
23/// sh:qualifiedValueShape of Q itself. The set of sibling shapes is empty otherwise."
24struct QualifiedValueShapeSiblings<RDF: FocusRDF> {
25    _marker: PhantomData<RDF>,
26    property_qualified_value_shape_path: SHACLPath,
27}
28
29impl<RDF> RDFNodeParse<RDF> for QualifiedValueShapeSiblings<RDF>
30where
31    RDF: FocusRDF,
32{
33    type Output = Vec<Object>;
34
35    fn parse_focused(&self, rdf: &mut RDF) -> Result<Self::Output, RDFError> {
36        match rdf.get_focus() {
37            Some(focus) => {
38                let mut siblings = Vec::new();
39                let maybe_disjoint =
40                    rdf.object_for(focus, &ShaclVocab::sh_qualified_value_shapes_disjoint().clone().into())?;
41                if let Some(disjoint) = maybe_disjoint {
42                    match disjoint {
43                        Object::Literal(ConcreteLiteral::BooleanLiteral(true)) => {
44                            let qvs = rdf.objects_for(focus, &ShaclVocab::sh_qualified_value_shape().clone().into())?;
45                            if !qvs.is_empty() {
46                                let ps = rdf.subjects_for(&ShaclVocab::sh_property().clone().into(), focus)?;
47                                for property_parent in ps {
48                                    let candidate_siblings = rdf.objects_for_shacl_path(
49                                        &property_parent,
50                                        &self.property_qualified_value_shape_path,
51                                    )?;
52                                    for sibling in candidate_siblings {
53                                        if !qvs.contains(&sibling) {
54                                            let sibling_node = RDF::term_as_object(&sibling)?;
55                                            siblings.push(sibling_node);
56                                        }
57                                    }
58                                }
59                            }
60                        },
61                        Object::Literal(ConcreteLiteral::BooleanLiteral(false)) => {},
62                        _ => { /* TODO - Raise error */ },
63                    }
64                }
65                Ok(siblings)
66            },
67            None => Err(RDFError::NoFocusNodeError),
68        }
69    }
70}

72pub(crate) fn qualified_value_shape<RDF: FocusRDF>() -> impl RDFNodeParse<RDF, Output = Vec<Component>> {
73    ObjectsPropertyParser::new(ShaclVocab::sh_qualified_value_shape().clone()).then(|qvs| {
74        let set = qvs.into_iter().collect();
75        parse_qualified_value_shape::<RDF>(set)
76    })
77}

79fn qualified_value_shape_disjoint_parser<RDF: FocusRDF>() -> impl RDFNodeParse<RDF, Output = Option<bool>> {
80    SingleBoolPropertyParser::new(ShaclVocab::sh_qualified_value_shapes_disjoint().clone()).optional()
81}

83fn qualified_min_count_parser<RDF: FocusRDF>() -> impl RDFNodeParse<RDF, Output = Option<isize>> {
84    SingleIntegerPropertyParser::new(ShaclVocab::sh_qualified_min_count().clone()).optional()
85}

87fn qualified_max_count_parser<RDF: FocusRDF>() -> impl RDFNodeParse<RDF, Output = Option<isize>> {
88    SingleIntegerPropertyParser::new(ShaclVocab::sh_qualified_max_count().clone()).optional()
89}

91fn qualified_value_shape_siblings<RDF: FocusRDF>() -> QualifiedValueShapeSiblings<RDF> {
92    QualifiedValueShapeSiblings {
93        _marker: PhantomData,
94        property_qualified_value_shape_path: SHACLPath::sequence(vec![
95            SHACLPath::iri(ShaclVocab::sh_property().clone()),
96            SHACLPath::iri(ShaclVocab::sh_qualified_value_shape().clone()),
97        ]),
98    }
99}
100
101fn parse_qualified_value_shape<RDF: FocusRDF>(qvs: HashSet<Object>) -> impl RDFNodeParse<RDF, Output = Vec<Component>> {
102    qualified_value_shape_disjoint_parser()
103        .and(qualified_min_count_parser())
104        .and(qualified_max_count_parser())
105        .and(qualified_value_shape_siblings())
106        .flat_map(move |(((maybe_disjoint, maybe_mins), maybe_maxs), siblings)| {
107            Ok(build_qualified_shape(
108                qvs.clone(),
109                maybe_disjoint,
110                maybe_mins,
111                maybe_maxs,
112                siblings,
113            ))
114        })
115}

117fn build_qualified_shape(
118    terms: HashSet<Object>,
119    disjoint: Option<bool>,
120    q_min_count: Option<isize>,
121    q_max_count: Option<isize>,
122    siblings: Vec<Object>,
123) -> Vec<Component> {
124    let mut result = Vec::new();
125    for term in terms {
126        let shape = Component::QualifiedValueShape {
127            shape: term.clone(),
128            q_min_count,
129            q_max_count,
130            disjoint,
131            siblings: siblings.clone(),
132        };
133        result.push(shape);
134    }
135    result
