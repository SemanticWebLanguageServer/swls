11pub(crate) fn property_shape<RDF: FocusRDF + 'static>() -> impl RDFNodeParse<RDF, Output = PropertyShape<RDF>> {
12    FocusParser::new().then(move |focus: RDF::Term| {
13        HasTypeParser::new(ShaclVocab::sh_property_shape().clone())
14            .optional()
15            .with(
16                ObjectParser::new()
17                    .and(path())
18                    .then(move |(id, path)| SuccessParser::new(PropertyShape::new(id, path))),
19            )
20            .then(move |ps| SetFocusParser::new(focus.clone()).with(SuccessParser::new(ps)))
21            .then(|ps| {
22                severity()
23                    .optional()
24                    .flat_map(move |sev| Ok(ps.clone().with_severity(sev)))
25            })
26            .then(|ps| reifier_shape().flat_map(move |r_shape| Ok(ps.clone().with_reifier_shape(r_shape))))
27            .then(|ps| targets().flat_map(move |ts| Ok(ps.clone().with_targets(ts))))
28            .then(|ps| {
29                property()
30                    .flat_map(move |prop_shapes| Ok(ps.clone().with_property_shapes(prop_shapes)))
31                    .then(move |ps_with_props| property_shape_components(ps_with_props))
32            })
33    })
34}

36fn property_shape_components<RDF: FocusRDF>(
37    ps: PropertyShape<RDF>,
38) -> impl RDFNodeParse<RDF, Output = PropertyShape<RDF>> {
39    components().flat_map(move |cs| Ok(ps.clone().with_components(cs)))
