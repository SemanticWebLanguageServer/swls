19impl LiteralHelper {
20    pub fn to_lit(self, (value, quote_style): (String, StringStyle)) -> RDFLiteral {
21        match self {
22            LiteralHelper::LangTag(lang) => RDFLiteral {
23                value,
24                quote_style,
25                lang: Some(lang),
26                ty: None,
27            },
28            LiteralHelper::DataType(dt) => RDFLiteral {
29                value,
30                quote_style,
31                lang: None,
32                ty: Some(dt),
33            },
34            LiteralHelper::None => RDFLiteral {
35                value,
36                quote_style,
37                lang: None,
38                ty: None,
39            },
40        }
41    }
42}

44fn literal() -> impl Parser<Token, Literal, Error = Simple<Token, S>> + Clone {
45    let lt = select! { Token::LangTag(x) => LiteralHelper::LangTag(x)};
46
47    let dt = named_node()
48        .then_ignore(just(Token::DataTypeDelim))
49        .map(|x| LiteralHelper::DataType(x));
50
51    lt.or(dt)
52        .or(empty().to(LiteralHelper::None))
53        .then(select! {
54            Token::Str(x, style) => (x, style)
55        })
56        .map(|(h, x)| h.to_lit(x))
57        .map(|x| Literal::RDF(x))
58        .or(select! {
59            Token::Number(x) => Literal::Numeric(x),
60            Token::True => Literal::Boolean(true),
61            Token::False => Literal::Boolean(false),
62        })
63}

65pub fn named_node() -> impl Parser<Token, NamedNode, Error = Simple<Token, S>> + Clone {
66    let invalid = select! {
67        Token::Invalid(_) => NamedNode::Invalid,
68    }
69    .validate(move |x, span: S, emit| {
70        emit(Simple::custom(
71            span.end..span.end,
72            format!("Expected a named node."),
73        ));
74        x
75    });
76
77    select! {
78        Token::PredType => NamedNode::A,
79        Token::IRIRef(x) => NamedNode::Full(x),
80        Token::PNameLN(x, b) => NamedNode::Prefixed { prefix: x.unwrap_or_default(), value: b },
81    }
82    .or(invalid)
83}

85pub fn is_term_like(token: &Token) -> bool {
86    match token {
87        // Token::True => true,
88        // Token::False => true,
89        // Token::IRIRef(_) => true,
90        // Token::PNameLN(_, _) => true,
91        // Token::BlankNodeLabel(_) => true,
92        // Token::Number(_) => true,
93        // Token::Str(_, _) => true,
94        // Token::ANON => true,
95        // Token::Null => true,
96        Token::Invalid(_) => true,
97        // Token::Variable(_) => true,
98        // Token::PrefixTag => true,
99        // Token::BaseTag => true,
100        // Token::SparqlPrefix => true,
101        // Token::SparqlBase => true,
102        _ => false,
103    }
104}

106pub fn expect_token(
107    token: Token,
108    valid: impl Fn(&Token) -> bool + Clone,
109) -> impl Parser<Token, Token, Error = Simple<Token, S>> + Clone {
110    let inner_token = token.clone();
111    just(token.clone()).or(none_of([token.clone()])
112        .rewind()
113        .try_map(move |t, span| {
114            if valid(&t) {
115                Ok(t)
116            } else {
117                Err(Simple::expected_input_found(
118                    span,
119                    [Some(inner_token.clone())],
120                    Some(t),
121                ))
122            }
123        })
124        .validate(move |_, span: S, emit| {
125            emit(Simple::expected_input_found(
126                span,
127                [Some(token.clone())],
128                None,
129            ));
130            token.clone()
131        }))
132}

134fn blank_node() -> impl Parser<Token, BlankNode, Error = Simple<Token>> + Clone {
135    recursive(|bn| {
136        po(bn)
137            .map_with_span(spanned)
138            .separated_by(just(Token::PredicateSplit))
139            .allow_leading()
140            .map(|mut x| {
141                x.reverse();
142                x
143            })
144            .delimited_by(just(Token::SqClose), just(Token::SqOpen))
145            .map(|x| BlankNode::Unnamed(x))
146            .or(select! {
147                Token::BlankNodeLabel(x) => BlankNode::Named(x),
148            })
149    })
150}

152fn subject() -> impl Parser<Token, Term, Error = Simple<Token, S>> + Clone {
153    let nn = named_node().map(|x| Term::NamedNode(x));
154    let bn = blank_node().map(|x| Term::BlankNode(x));
155    let var = variable().map(|x| Term::Variable(x));
156
157    nn.or(bn).or(var)
158}
159
160fn variable() -> impl Parser<Token, Variable, Error = Simple<Token, S>> + Clone {
161    select! {
162        Token::Variable(x) => Variable(x),
163    }
164}

166fn term(
167    bn: impl Clone + Parser<Token, BlankNode, Error = Simple<Token>> + 'static,
168) -> impl Parser<Token, Term, Error = Simple<Token>> + Clone {
169    recursive(|term| {
170        let collection = term
171            .map_with_span(spanned)
172            .repeated()
173            .map(|mut x| {
174                x.reverse();
175                x
176            })
177            .delimited_by(just(Token::BracketClose), just(Token::BracketOpen))
178            .map(|x| Term::Collection(x));
179
180        let nn = named_node().map(|x| Term::NamedNode(x));
181        let blank = bn.map(|x| Term::BlankNode(x));
182        let literal = literal().map(|x| Term::Literal(x));
183        let variable = variable().map(|x| Term::Variable(x));
184        collection.or(literal).or(nn).or(blank).or(variable)
185    })
186}

188fn po(
189    bn: impl Clone + Parser<Token, BlankNode, Error = Simple<Token>> + 'static,
190) -> impl Parser<Token, PO, Error = Simple<Token>> + Clone {
191    term(bn.clone())
192        .labelled("object")
193        .map_with_span(spanned)
194        .separated_by(just(Token::Comma))
195        .at_least(1)
196        .map(|mut x| {
197            x.reverse();
198            x
199        })
200        // .allow_leading() // TODO check in grammar
201        .then_with(move |os| {
202            let os1 = os.clone();
203            let predicate = term(bn.clone())
204                .map_with_span(spanned)
205                .map(move |pred| (os1.clone(), pred))
206                .labelled("basic predicate");
207
208            let os2 = os.clone();
209            // let end = os[0].span().end;
210            //
211
212            let alt_pred = one_of([Token::SqOpen, Token::PredicateSplit])
213                .rewind()
214                .validate(move |_, span: S, emit| {
215                    emit(Simple::custom(
216                        span.end..span.end,
217                        format!("Expected an object."),
218                    ));
219                    ()
220                })
221                .map(move |_| {
222                    let mut os = os2.clone();
223                    let pred = os[0].clone();
224                    os[0] = spanned(Term::NamedNode(NamedNode::Invalid), os[0].span().clone());
225
226                    (os, pred)
227                })
228                .labelled("alt predicate");
229
230            predicate.or(alt_pred)
231        })
232        .map(|(object, predicate)| PO { predicate, object })
233}

235pub fn triple() -> impl Parser<Token, Triple, Error = Simple<Token>> + Clone {
236    expect_token(Token::Stop, |_| true)
237        .ignore_then(
238            po(blank_node())
239                .map_with_span(spanned)
240                .separated_by(just(Token::PredicateSplit))
241                .allow_leading()
242                .map(|mut x| {
243                    x.reverse();
244                    x
245                })
246                .validate(|po, span: S, emit| {
247                    if po.is_empty() {
248                        emit(Simple::custom(
249                            span.clone(),
250                            format!("Expected at least one predicate object."),
251                        ));
252                        (
253                            vec![spanned(
254                                PO {
255                                    predicate: spanned(Term::Invalid, span.clone()),
256                                    object: vec![spanned(Term::Invalid, span.clone())],
257                                },
258                                span,
259                            )],
260                            false,
261                        )
262                    } else {
263                        (po, true)
264                    }
265                }),
266        )
267        .then_with(move |(po, succesful)| {
268            let po2 = po.clone();
269            let basic_subj = subject()
270                .map_with_span(spanned)
271                .map(move |subj| (po2.clone(), subj));
272
273            let end = po[0].span().end;
274            let start = if succesful {
275                empty().boxed()
276            } else {
277                any().map(|_| ()).boxed()
278            };
279            let alt_subj = start.validate(move |_, _: S, emit| {
280                emit(Simple::custom(end..end, format!("Expected a predicate.")));
281
282                let mut po = po.clone();
283                let first = po[0].value_mut();
284
285                let subj = first.predicate.clone();
286                first.predicate = first.object.pop().unwrap();
287
288                first.object.push(Spanned(
289                    Term::NamedNode(NamedNode::Invalid),
290                    first.predicate.span().clone(),
291                ));
292
293                // Subject::NamedNode(NamedNode::Invalid)
294                (po, subj)
295            });
296
297            basic_subj.or(alt_subj)
298        })
299        .map(|(po, subject)| Triple { subject, po })
300}

302fn base() -> impl Parser<Token, Base, Error = Simple<Token>> + Clone {
303    expect_token(Token::Stop, |_| true)
304        .ignore_then(named_node().map_with_span(spanned))
305        .then(just(Token::BaseTag).map_with_span(|_, s| s))
306        .map(|(x, s)| Base(s, x))
307}

309fn prefix() -> impl Parser<Token, TurtlePrefix, Error = Simple<Token>> {
310    expect_token(Token::Stop, |_| true)
311        .ignore_then(named_node().map_with_span(spanned))
312        .then(select! { |span| Token::PNameLN(x, _) => Spanned(x.unwrap_or_default(), span)})
313        .then(just(Token::PrefixTag).map_with_span(|_, s| s))
314        .map(|((value, prefix), span)| TurtlePrefix {
315            span,
316            prefix,
317            value,
318        })
319}

321// Makes it easier to handle parts that are not ordered
322enum Statement {
323    Base(Spanned<Base>),
324    Prefix(Spanned<TurtlePrefix>),
325    Triple(Spanned<Triple>),
326}
327
328pub fn turtle<'a>(
329    location: &'a lsp_types::Url,
330) -> impl Parser<Token, Turtle, Error = Simple<Token>> + 'a {
331    let base = base().map_with_span(spanned).map(|b| Statement::Base(b));
332    let prefix = prefix()
333        .map_with_span(spanned)
334        .map(|b| Statement::Prefix(b));
335    let triple = triple()
336        .map_with_span(spanned)
337        .map(|b| Statement::Triple(b));
338
339    let statement = base.or(prefix).or(triple);
340    statement
341        .repeated()
342        .map(|statements| {
343            let mut base = None;
344            let mut prefixes = Vec::new();
345            let mut triples = Vec::new();
346            for statement in statements {
347                match statement {
348                    Statement::Base(b) => base = Some(b),
349                    Statement::Prefix(p) => prefixes.push(p),
350                    Statement::Triple(t) => triples.push(t),
351                }
352            }
353            prefixes.reverse();
354            triples.reverse();
355
356            Turtle::new(base, prefixes, triples, location)
357        })
358        .then_ignore(end())
359}

361pub fn parse_turtle(
362    location: &lsp_types::Url,
363    tokens: Vec<Spanned<Token>>,
364    len: usize,
365) -> (Spanned<Turtle>, Vec<(usize, Simple<Token>)>) {
366    let rev_range = |range: std::ops::Range<usize>| (len - range.end)..(len - range.start);
367    let stream = chumsky::Stream::from_iter(
368        0..len,
369        tokens
370            .into_iter()
371            .rev()
372            .filter(|x| !x.is_comment())
373            .map(|Spanned(x, s)| (x, rev_range(s))),
374    );
375
376    let parser = turtle(location)
377        .map_with_span(spanned)
378        .then_ignore(end().recover_with(skip_then_retry_until([])));
379
380    info!("Parsing {}", location.as_str());
381    let (mut json, json_errors) = parser.parse_recovery(stream);
382
383    json.iter_mut().for_each(|turtle| turtle.0.fix_spans(len));
384
385    let json_errors: Vec<_> = json_errors.into_iter().map(|error| (len, error)).collect();
386    if !json_errors.is_empty() {
387        info!("Errors");
388        for e in &json_errors {
389            info!("Error {:?}", e);
390        }
391    }
392
393    (
394        json.unwrap_or(Spanned(Turtle::empty(location), 0..len)),
395        json_errors,
396    )
397}

412    pub fn parse_it<T, P: Parser<Token, T, Error = Simple<Token>>>(
413        turtle: &str,
414        parser: P,
415    ) -> (Option<T>, Vec<Simple<Token>>) {
416        let tokens = tokenizer::parse_tokens().parse(turtle).unwrap();
417        let end = turtle.len()..turtle.len();
418        let stream = Stream::from_iter(
419            end,
420            tokens
421                .into_iter()
422                .map(|Spanned(x, y)| (x, y))
423                .rev()
424                .filter(|x| !x.0.is_comment()),
425        );
426
427        parser.parse_recovery(stream)
428    }

442                .rev()
443                .filter(|x| !x.0.is_comment()),
