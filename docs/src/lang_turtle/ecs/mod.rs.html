13pub fn setup_parsing(world: &mut World) {
14    use lsp_core::feature::parse::*;
15    world.schedule_scope(ParseLabel, |_, schedule| {
16        schedule.add_systems((
17            parse_source,
18            parse_turtle_system.after(parse_source),
19            derive_prefixes.after(parse_turtle_system).before(prefixes),
20            derive_triples.after(parse_turtle_system).before(triples),

22    });
23}

41fn derive_prefixes(
42    query: Query<(Entity, &Label, &Element<TurtleLang>), Changed<Element<TurtleLang>>>,
43    mut commands: Commands,
44) {
45    for (entity, url, turtle) in &query {
46        let prefixes: Vec<_> = turtle
47            .prefixes
48            .iter()
49            .flat_map(|prefix| {
50                let url = prefix.value.expand(turtle.value())?;
51                let url = lsp_types::Url::parse(&url).ok()?;
52                Some(Prefix {
53                    url,
54                    prefix: prefix.prefix.value().clone(),
55                })
56            })
57            .collect();
58
59        let base = turtle
60            .base
61            .as_ref()
62            .and_then(|b| {
63                b.0 .1
64                    .expand(turtle.value())
65                    .and_then(|x| lsp_types::Url::parse(&x).ok())
66            })
67            .unwrap_or(url.0.clone());
68
69        commands.entity(entity).insert(Prefixes(prefixes, base));
70    }
71}

84    #[test]
85    fn diagnostics_work() {
86        let (mut world, mut rx) = setup_world(TestClient::new(), crate::setup_world);
87
88        let t1 = "
89@prefix foaf: <>.
90            ";
91
92        let t2 = "
93@prefix foaf: <>.
94foaf:foaf
95            ";
96
97        let t3 = "
98@prefix foaf: <>.
99foa
100            ";
101
102        let entity = create_file(&mut world, t1, "http://example.com/ns#", "turtle", Open);
103        world.run_schedule(ParseLabel);
104        world.run_schedule(DiagnosticsLabel);
105
106        let mut get_diagnostics = move || {
107            let mut out: Vec<DiagnosticItem> = Vec::new();
108            while let Ok(Some(x)) = rx.try_next() {
109                out.push(x);
110            }
111            out
112        };
113        let items = get_diagnostics();
114        assert!(items[0].diagnostics.is_empty());
115
116        world
117            .entity_mut(entity)
118            .insert((Source(t2.to_string()), RopeC(Rope::from_str(t2))));
119        world.run_schedule(ParseLabel);
120        world.run_schedule(DiagnosticsLabel);
121
122        let items = get_diagnostics();
123        assert_eq!(items.len(), 2);
124        assert_eq!(items[0].diagnostics.len(), 2);
125        world
126            .entity_mut(entity)
127            .insert((Source(t3.to_string()), RopeC(Rope::from_str(t2))));
128        world.run_schedule(ParseLabel);
129        world.run_schedule(DiagnosticsLabel);
130
131        let items = get_diagnostics();
132        assert_eq!(items.len(), 2);
133        assert_eq!(items[0].diagnostics.len(), 5);
134    }
