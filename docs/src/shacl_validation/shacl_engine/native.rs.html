22pub struct NativeEngine {

44impl<S: NeighsRDF + Debug + 'static> Engine<S> for NativeEngine {
45    fn build_indexes(&mut self, store: &S) -> Result<(), Box<ValidateError>> {
46        self.class_index = Some(ClassIndex::build::<S>(store)?);
47        Ok(())
48    }
49
50    fn evaluate(
51        &mut self,
52        store: &S,
53        shape: &ShapeIR,
54        component: &ComponentIR,
55        value_nodes: &ValueNodes<S>,
56        source_shape: Option<&ShapeIR>,
57        maybe_path: Option<SHACLPath>,
58        shapes_graph: &SchemaIR,
59    ) -> Result<Vec<ValidationResult>, Box<ValidateError>> {
60        tracing::debug!("evaluate [NativeEngine] with shape {}", shape.id());
61        let shacl_component = ShaclComponent::new(component);
62        let validator = shacl_component.deref();
63        let result = validator
64            .validate_native(
65                component,
66                shape,
67                store,
68                self,
69                value_nodes,
70                source_shape,
71                maybe_path,
72                shapes_graph,
73            )
74            .map_err(|e| {
75                Box::new(ValidateError::ConstraintError {
76                    component: component.to_string(),
77                    source: e,
78                })
79            })?;
80        Ok(result)
81    }
82
83    /// https://www.w3.org/TR/shacl/#targetNode
84    fn target_node(&self, _: &S, node: &Object) -> Result<FocusNodes<S>, Box<ValidateError>> {
85        let node: S::Term = node.clone().into();
86        if node.is_blank_node() {
87            Err(Box::new(ValidateError::TargetNodeBlankNode))
88        } else {
89            Ok(FocusNodes::single(node.clone()))
90        }
91    }
92
93    fn target_class(&self, store: &S, class: &Object) -> Result<FocusNodes<S>, Box<ValidateError>> {
94        // use the pre-built class index (O(1) lookup)
95        if let Some(index) = &self.class_index {
96            let focus_nodes = index.instances_of(class).map(|obj| -> S::Term { obj.clone().into() });
97            return Ok(FocusNodes::from_iter(focus_nodes));
98        }
99
100        // Fallback: full graph scan (for backwards compatibility if index wasn't built)
101        let cls: S::Term = class.clone().into();
102        let focus_nodes = store
103            .shacl_instances_of(&cls)
104            .map_err(|e| ValidateError::TargetClassError {
105                msg: format!("Failed to get instances of class {class}: {e}"),
106            })?
107            .map(|subj| S::subject_as_term(&subj));
108
109        Ok(FocusNodes::from_iter(focus_nodes))
110    }
111
112    fn target_subject_of(&self, store: &S, predicate: &IriS) -> Result<FocusNodes<S>, Box<ValidateError>> {
113        let pred: S::IRI = predicate.clone().into();
114        let subjects = store
115            .triples_with_predicate(&pred)
116            .map_err(|_| ValidateError::SRdf)?
117            .map(Triple::into_subject)
118            .map(Into::into);
119        let focus_nodes = FocusNodes::from_iter(subjects);
120        Ok(focus_nodes)
121    }
122
123    fn target_object_of(&self, store: &S, predicate: &IriS) -> Result<FocusNodes<S>, Box<ValidateError>> {
124        let pred: S::IRI = predicate.clone().into();
125        let objects = store
126            .triples_with_predicate(&pred)
127            .map_err(|_| ValidateError::SRdf)?
128            .map(Triple::into_object);
129        Ok(FocusNodes::from_iter(objects))
130    }
131
132    fn implicit_target_class(&self, store: &S, subject: &Object) -> Result<FocusNodes<S>, Box<ValidateError>> {
133        // use the pre-built class index (O(1) lookup)
134        if let Some(index) = &self.class_index {
135            let instances = index.instances_of_with_subclasses(subject);
136            let focus_nodes = instances.into_iter().map(|obj| -> S::Term { obj.into() });
137            return Ok(FocusNodes::from_iter(focus_nodes));
138        }
139
140        // Fallback: full graph scan (for backwards compatibility if index wasn't built)
141        let term: S::Term = subject.clone().into();
142        let targets = store
143            .subjects_for(&RdfVocab::rdf_type().clone().into(), &term)
144            .map_err(|e| ValidateError::InstanceOf {
145                term: term.to_string(),
146                error: e.to_string(),
147            })?;
148
149        let subclass_targets = store
150            .subjects_for(&RdfsVocab::rdfs_subclass_of_str().clone().into(), &term)
151            .map_err(|e| ValidateError::SubClassOf {
152                term: term.to_string(),
153                error: e.to_string(),
154            })?
155            .into_iter()
156            .flat_map(move |subclass| {
157                store
158                    .subjects_for(&RdfVocab::rdf_type().clone().into(), &subclass)
159                    .map_err(|e| ValidateError::SubClassOf {
160                        term: subclass.to_string(),
161                        error: e.to_string(),
162                    })
163                    .into_iter()
164                    .flatten()
165            });
166
167        Ok(FocusNodes::from_iter(targets.into_iter().chain(subclass_targets)))
168    }
169
170    fn record_validation(&mut self, node: Object, shape_idx: ShapeLabelIdx, results: Vec<ValidationResult>) {
171        self.cache.record(node, shape_idx, results);
172    }
173
174    fn has_validated(&self, node: &Object, shape_idx: ShapeLabelIdx) -> bool {
175        self.cache.has_validated(node, shape_idx)
176    }
177
178    fn get_cached_results(&self, node: &Object, shape_idx: ShapeLabelIdx) -> Option<&Vec<ValidationResult>> {
179        self.cache.get_results(node, shape_idx)
180    }
