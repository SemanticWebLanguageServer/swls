15pub trait Engine<S: NeighsRDF> {
16    /// Pre-builds internal indexes from the data graph for faster target resolution.
17    ///
18    /// This should be called **once** before the validation loop starts.
19    fn build_indexes(&mut self, _store: &S) -> Result<(), Box<ValidateError>> {
20        Ok(())
21    }
22
23    #[allow(clippy::too_many_arguments)]
24    fn evaluate(
25        &mut self,
26        store: &S,
27        shape: &ShapeIR,
28        component: &ComponentIR,
29        value_nodes: &ValueNodes<S>,
30        source_shape: Option<&ShapeIR>,
31        maybe_path: Option<SHACLPath>,
32        shapes_graph: &SchemaIR,
33    ) -> Result<Vec<ValidationResult>, Box<ValidateError>>;
34
35    fn focus_nodes(&self, store: &S, targets: &[CompiledTarget]) -> Result<FocusNodes<S>, Box<ValidateError>> {
36        let targets_iter: Vec<FocusNodes<S>> = targets
37            .iter()
38            .flat_map(|target| match target {
39                CompiledTarget::Node(node) => self.target_node(store, node),
40                CompiledTarget::Class(class) => self.target_class(store, class),
41                CompiledTarget::SubjectsOf(predicate) => self.target_subject_of(store, predicate),
42                CompiledTarget::ObjectsOf(predicate) => self.target_object_of(store, predicate),
43                CompiledTarget::ImplicitClass(node) => self.implicit_target_class(store, node),
44                CompiledTarget::WrongTargetNode(_) => todo!(),
45                CompiledTarget::WrongTargetClass(_) => todo!(),
46                CompiledTarget::WrongSubjectsOf(_) => todo!(),
47                CompiledTarget::WrongObjectsOf(_) => todo!(),
48                CompiledTarget::WrongImplicitClass(_) => todo!(),
49            })
50            .collect();
51        let ts = targets_iter.into_iter().flatten();
52        Ok(FocusNodes::from_iter(ts))
53    }
54
55    /// If s is a shape in a shapes graph SG and s has value t for sh:targetNode
56    /// in SG then { t } is a target from any data graph for s in SG.
57    fn target_node(&self, store: &S, node: &Object) -> Result<FocusNodes<S>, Box<ValidateError>>;
58
59    fn target_class(&self, store: &S, class: &Object) -> Result<FocusNodes<S>, Box<ValidateError>>;
60
61    fn target_subject_of(&self, store: &S, predicate: &IriS) -> Result<FocusNodes<S>, Box<ValidateError>>;
62
63    fn target_object_of(&self, store: &S, predicate: &IriS) -> Result<FocusNodes<S>, Box<ValidateError>>;
64
65    fn implicit_target_class(&self, store: &S, shape: &Object) -> Result<FocusNodes<S>, Box<ValidateError>>;
66
67    fn path(
68        &self,
69        store: &S,
70        shape: &PropertyShapeIR,
71        focus_node: &S::Term,
72    ) -> Result<FocusNodes<S>, Box<ValidateError>> {
73        let nodes =
74            store
75                .objects_for_shacl_path(focus_node, shape.path())
76                .map_err(|e| ValidateError::ObjectsSHACLPath {
77                    focus_node: focus_node.to_string(),
78                    shacl_path: shape.path().to_string(),
79                    error: e.to_string(),
80                })?;
81        Ok(FocusNodes::new(nodes))
82    }
83
84    fn record_validation(&mut self, node: Object, shape_idx: ShapeLabelIdx, results: Vec<ValidationResult>);
85
86    fn has_validated(&self, node: &Object, shape_idx: ShapeLabelIdx) -> bool;
87
88    /// Returns the cached validation results for a given `(node, shape_idx)` pair, if any.
89    fn get_cached_results(&self, node: &Object, shape_idx: ShapeLabelIdx) -> Option<&Vec<ValidationResult>>;
90}
