41impl<S: QueryRDF + NeighsRDF + Debug + 'static> Engine<S> for SparqlEngine {
42    fn evaluate(
43        &mut self,
44        store: &S,
45        shape: &ShapeIR,
46        component: &ComponentIR,
47        value_nodes: &ValueNodes<S>,
48        source_shape: Option<&ShapeIR>,
49        maybe_path: Option<SHACLPath>,
50        shape_graph: &SchemaIR,
51    ) -> Result<Vec<ValidationResult>, Box<ValidateError>> {
52        let shacl_component = ShaclComponent::new(component);
53        let validator = shacl_component.deref();
54        let result = validator
55            .validate_sparql(
56                component,
57                shape,
58                store,
59                value_nodes,
60                source_shape,
61                maybe_path,
62                shape_graph,
63            )
64            .map_err(|e| {
65                Box::new(ValidateError::ConstraintError {
66                    component: component.to_string(),
67                    source: e,
68                })
69            })?;
70        Ok(result)
71    }

73    /// If s is a shape in a shapes graph SG and s has value t for sh:targetNode
74    /// in SG then { t } is a target from any data graph for s in SG.
75    fn target_node(&self, store: &S, node: &Object) -> Result<FocusNodes<S>, Box<ValidateError>> {
76        let node: S::Term = node.clone().into();
77        if node.is_blank_node() {
78            return Err(Box::new(ValidateError::TargetNodeBlankNode));
79        }
80
81        let query = formatdoc! {"
82            SELECT DISTINCT ?this
83            WHERE {{
84                BIND ({} AS ?this)
85            }}
86        ", node};
87
88        select(store, query, "this").map_err(|e| {
89            Box::new(ValidateError::SparqlError {
90                msg: "target_node".to_string(),
91                source: e,
92            })
93        })?;
94
95        Err(Box::new(ValidateError::NotImplemented {
96            msg: "target_node".to_string(),
97        }))
98    }

100    fn target_class(&self, store: &S, class: &Object) -> Result<FocusNodes<S>, Box<ValidateError>> {
101        let class: S::Term = class.clone().into();
102        if !class.is_iri() {
103            return Err(Box::new(ValidateError::TargetClassNotIri));
104        }
105
106        let query = formatdoc! {"
107            PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
108            PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
109
110            SELECT DISTINCT ?this
111            WHERE {{
112                ?this rdf:type/rdfs:subClassOf* {} .
113            }}
114        ", class};
115
116        select(store, query, "this").map_err(|e| {
117            Box::new(ValidateError::SparqlError {
118                msg: "target_class".to_string(),
119                source: e,
120            })
121        })?;
122
123        Err(Box::new(ValidateError::NotImplemented {
124            msg: "target_class".to_string(),
125        }))
126    }
