18impl<R: NeighsRDF + Debug + 'static> NativeValidator<R> for LessThan {
19    fn validate_native(
20        &self,
21        component: &ComponentIR,
22        shape: &ShapeIR,
23        store: &R,
24        _engine: &mut dyn Engine<R>,
25        value_nodes: &ValueNodes<R>,
26        _source_shape: Option<&ShapeIR>,
27        maybe_path: Option<SHACLPath>,
28        _shapes_graph: &SchemaIR,
29    ) -> Result<Vec<ValidationResult>, ConstraintError> {
30        let mut validation_results = Vec::new();
31        let component = Object::iri(component.into());
32
33        for (focus_node, nodes) in value_nodes.iter() {
34            let subject: R::Subject = <R as Rdf>::term_as_subject(focus_node).unwrap();
35            let iri_owned: R::IRI = self.iri().clone().into();
36            match store.triples_with_subject_predicate(&subject, &iri_owned) {
37                Ok(triples_iter) => {
38                    // Collect nodes to compare
39                    for triple in triples_iter {
40                        let value = triple.obj();
41                        let node1 = <R as Rdf>::term_as_object(value).unwrap();
42                        for value2 in nodes.iter() {
43                            let node2 = <R as Rdf>::term_as_object(value2).unwrap();
44                            let message = match node2.partial_cmp(&node1) {
45                                None => Some(format!(
46                                    "LessThan constraint violated: {node1} is not comparable to {node2}"
47                                )),
48                                Some(ord) if ord.is_ge() => Some(format!(
49                                    "LessThan constraint violated: {node1} is not less than {node2}"
50                                )),
51                                _ => None,
52                            };
53                            if let Some(msg) = message {
54                                let validation_result =
55                                    ValidationResult::new(shape.id().clone(), component.clone(), shape.severity())
56                                        .with_message(msg.as_str())
57                                        .with_path(maybe_path.clone());
58                                validation_results.push(validation_result);
59                            }
60                        }
61                    }
62                },
63                Err(e) => {
64                    let message = format!(
65                        "LessThan: Error trying to find triples for subject {} and predicate {}: {e}",
66                        subject,
67                        self.iri()
68                    );
69                    let validation_result =
70                        ValidationResult::new(shape.id().clone(), component.clone(), shape.severity())
71                            .with_message(message.as_str())
72                            .with_path(maybe_path.clone());
73                    validation_results.push(validation_result);
74                },
75            };
76        }
77        Ok(validation_results)
78    }
79}
