1use rudof_rdf::rdf_core::NeighsRDF;
2use rudof_rdf::rdf_core::term::{Object, Triple};
3use rudof_rdf::rdf_core::vocabs::{RdfVocab, RdfsVocab};
4use std::collections::{HashMap, HashSet};
5
6use crate::validate_error::ValidateError;
7
8/// Pre-computed inverted index for class-based target resolution.
9///
10/// Instead of scanning all `rdf:type` triples every time `target_class` or
11/// `implicit_target_class` is called, this index is built **once** before
12/// validation starts, turning the per-shape O(n) scan into a single O(n)
13/// build step followed by O(1) lookups per class.
14///
15/// The index stores:
16/// - `class_instances`: `class_object → { subject_as_term, … }` built from all
17///   `?s rdf:type ?class` triples.
18/// - `subclass_map`: `class_object → { subclass_as_term, … }` built from all
19///   `?sub rdfs:subClassOf ?class` triples (one level).
20#[derive(Debug, Clone, Default)]
21pub struct ClassIndex {
22    /// Maps each class to the set of its direct `rdf:type` instances.
23    class_instances: HashMap<Object, HashSet<Object>>,
24    /// Maps each class to the set of its direct `rdfs:subClassOf` subclasses.
25    subclass_map: HashMap<Object, HashSet<Object>>,
26}
27
28impl ClassIndex {
29    /// Builds both indexes by performing a single pass over all triples.
30    pub fn build<S: NeighsRDF>(store: &S) -> Result<Self, Box<ValidateError>> {
31        let mut class_instances: HashMap<Object, HashSet<Object>> = HashMap::new();
32        let mut subclass_map: HashMap<Object, HashSet<Object>> = HashMap::new();
33
34        let rdf_type: S::IRI = RdfVocab::rdf_type().clone().into();
35        let rdfs_subclass_of: S::IRI = RdfsVocab::rdfs_subclass_of_str().clone().into();
36
37        for triple in store
38            .triples()
39            .map_err(|e| ValidateError::ClassIndexBuild { error: e.to_string() })?
40        {
41            let (subj, pred, obj) = triple.into_components();
42
43            if pred == rdf_type {
44                // ?subj rdf:type ?obj  →  obj is a class, subj is an instance
45                let class_obj = S::term_as_object(&obj).ok();
46                let instance_obj = S::term_as_object(&S::subject_as_term(&subj)).ok();
47                if let (Some(cls), Some(inst)) = (class_obj, instance_obj) {
48                    class_instances.entry(cls).or_default().insert(inst);
49                }
50            } else if pred == rdfs_subclass_of {
51                // ?subj rdfs:subClassOf ?obj  →  subj is subclass of obj
52                let parent_obj = S::term_as_object(&obj).ok();
53                let child_obj = S::term_as_object(&S::subject_as_term(&subj)).ok();
54                if let (Some(parent), Some(child)) = (parent_obj, child_obj) {
55                    subclass_map.entry(parent).or_default().insert(child);
56                }
57            }
58        }
59
60        Ok(ClassIndex {
61            class_instances,
62            subclass_map,
63        })
64    }
65
66    /// Returns the set of direct instances of the given class.
67    pub fn instances_of(&self, class: &Object) -> impl Iterator<Item = &Object> {
68        self.class_instances.get(class).into_iter().flat_map(|set| set.iter())
69    }
70
71    /// Returns the set of direct subclasses of the given class.
72    pub fn subclasses_of(&self, class: &Object) -> impl Iterator<Item = &Object> {
73        self.subclass_map.get(class).into_iter().flat_map(|set| set.iter())
74    }
75
76    /// Returns instances of the class and instances of all its direct subclasses.
77    pub fn instances_of_with_subclasses(&self, class: &Object) -> HashSet<Object> {
78        let mut result: HashSet<Object> = self.instances_of(class).cloned().collect();
79        for subclass in self.subclasses_of(class) {
80            for instance in self.instances_of(subclass) {
81                result.insert(instance.clone());
82            }
83        }
84        result
85    }
86}
