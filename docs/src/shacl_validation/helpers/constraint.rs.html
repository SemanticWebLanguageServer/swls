12fn apply<S: Rdf, I: IterationStrategy<S>>(
13    component: &ComponentIR,
14    shape: &ShapeIR,
15    value_nodes: &ValueNodes<S>,
16    iteration_strategy: I,
17    evaluator: impl Fn(&I::Item) -> Result<bool, ConstraintError>,
18    message: &str,
19    maybe_path: Option<SHACLPath>,
20) -> Result<Vec<ValidationResult>, ConstraintError> {
21    let results = iteration_strategy
22        .iterate(value_nodes)
23        .flat_map(|(focus_node, item)| {
24            let focus = S::term_as_object(focus_node).ok()?;
25            let component = Object::iri(component.into());
26            let shape_id = shape.id();
27            let source = Some(shape_id);
28            let value = iteration_strategy.to_object(item);
29            if let Ok(condition) = evaluator(item)
30                && condition
31            {
32                return Some(
33                    ValidationResult::new(focus, component, shape.severity())
34                        .with_source(source.cloned())
35                        .with_message(message)
36                        .with_path(maybe_path.clone())
37                        .with_value(value),
38                );
39            }
40            None
41        })
42        .collect();
43
44    Ok(results)
45}

47fn apply_with_focus<S: Rdf, I: IterationStrategy<S>>(
48    component: &ComponentIR,
49    shape: &ShapeIR,
50    value_nodes: &ValueNodes<S>,
51    iteration_strategy: I,
52    evaluator: impl Fn(&S::Term, &I::Item) -> Result<bool, ConstraintError>,
53    message: &str,
54    maybe_path: Option<SHACLPath>,
55) -> Result<Vec<ValidationResult>, ConstraintError> {
56    let results = iteration_strategy
57        .iterate(value_nodes)
58        .flat_map(|(focus_node, item)| {
59            let focus = S::term_as_object(focus_node).ok()?;
60            let component = Object::iri(component.into());
61            let shape_id = shape.id();
62            let source = Some(shape_id);
63            let value = iteration_strategy.to_object(item);
64            match evaluator(focus_node, item) {
65                Ok(true) => Some(
66                    ValidationResult::new(focus, component, shape.severity())
67                        .with_source(source.cloned())
68                        .with_message(message)
69                        .with_path(maybe_path.clone())
70                        .with_value(value),
71                ),
72                Ok(false) => None,
73                Err(err) => {
74                    debug!("LessThan.validate_native with focus: {:?}, err: {err}", focus);
75                    None
76                },
77            }
78        })
79        .collect();
80
81    Ok(results)
82}

84/// Validate with a boolean evaluator. If the evaluator returns true, it means that there is a violation
85pub fn validate_with<S: Rdf, I: IterationStrategy<S>>(
86    component: &ComponentIR,
87    shape: &ShapeIR,
88    value_nodes: &ValueNodes<S>,
89    iteration_strategy: I,
90    evaluator: impl Fn(&I::Item) -> bool,
91    message: &str,
92    maybe_path: Option<SHACLPath>,
93) -> Result<Vec<ValidationResult>, ConstraintError> {
94    apply(
95        component,
96        shape,
97        value_nodes,
98        iteration_strategy,
99        |item: &I::Item| Ok(evaluator(item)),
100        message,
101        maybe_path,
102    )
103}

126pub fn validate_ask_with<S: QueryRDF>(
127    component: &ComponentIR,
128    shape: &ShapeIR,
129    store: &S,
130    value_nodes: &ValueNodes<S>,
131    eval_query: impl Fn(&S::Term) -> String,
132    message: &str,
133    maybe_path: Option<SHACLPath>,
134) -> Result<Vec<ValidationResult>, ConstraintError> {
135    apply(
136        component,
137        shape,
138        value_nodes,
139        ValueNodeIteration,
140        |value_node| match store.query_ask(&eval_query(value_node)) {
141            Ok(ask) => Ok(!ask),
142            Err(err) => Err(ConstraintError::Query(format!("ASK query failed: {err}"))),
143        },
144        message,
145        maybe_path,
146    )
