44pub fn find_annotation(
45    annotations: &Option<Vec<Annotation>>,
46    predicate: &IriS,
47    prefixmap: &PrefixMap,
48) -> std::result::Result<Option<ObjectValue>, PrefixMapError> {
49    if let Some(anns) = annotations {
50        for a in anns.iter() {
51            let iri_predicate = prefixmap.resolve_iriref(&a.predicate())?;
52            if *predicate == iri_predicate {
53                return Ok(Some(a.object()));
54            }
55        }
56        Ok(None)
57    } else {
58        Ok(None)
59    }
60}
