30/// A SPARQL endpoint client that implements the SRDF interface.
31///
32/// This struct provides methods for querying SPARQL endpoints with support for
33/// different query types (SELECT, CONSTRUCT, ASK) and result formats.
34///
35/// # Performance
36///
37/// - Uses `Arc` for shared data (clients, prefix map) to enable cheap cloning
38/// - Pre-allocates collections when size is known
39/// - Caches HTTP clients with appropriate headers for each format
40#[derive(Debug, Clone)]
41pub struct SparqlEndpoint {
42    /// The IRI of the SPARQL endpoint.
43    endpoint_iri: IriS,
44
45    /// Prefix map for qualifying IRIs.
46    prefixmap: Arc<PrefixMap>,
47
48    /// HTTP client configured for SELECT queries (expects JSON results).
49    client: Arc<reqwest::Client>,
50
51    /// Cache of HTTP clients for CONSTRUCT queries
52    construct_clients: Arc<RwLock<HashMap<QueryResultFormat, Arc<reqwest::Client>>>>,
53}
54
55impl PartialEq for SparqlEndpoint {
56    /// Two endpoints are equal if they have the same IRI.
57    ///
58    /// Note: This compares only the endpoint IRI, not the prefix maps or clients.
59    fn eq(&self, other: &Self) -> bool {
60        self.endpoint_iri == other.endpoint_iri
61    }
62}

196    /// Executes a SPARQL SELECT query asynchronously.
197    ///

205    ///

224    ///

226    ///

229    ///
230    ///     Ok(())
231    /// }

233    pub async fn query_select_async(&self, query: &str) -> Result<QuerySolutions<Self>> {
234        let solutions = make_sparql_query_select_async(query, &self.client, &self.endpoint_iri).await?;
235
236        // Pre-allocate with known capacity for better performance
237        let mut qs = Vec::with_capacity(solutions.len());
238        for solution in &solutions {
239            qs.push(cnv_query_solution(solution));
240        }
241
242        Ok(QuerySolutions::new(qs, (*self.prefixmap).clone()))
243    }

264    ///
265    /// # Examples
266    ///
267    /// ```no_run

269    /// use rudof_rdf::rdf_core::query::QueryResultFormat;

272    /// async fn main() -> Result<(), Box<dyn std::error::Error>> {
273    ///     let endpoint = SparqlEndpoint::wikidata()?;
274    ///     let query = "CONSTRUCT { ?s ?p ?o } WHERE { ?s ?p ?o } LIMIT 10";
275    ///
276    ///     let turtle = endpoint
277    ///         .query_construct_async(query, &QueryResultFormat::Turtle)
278    ///         .await?;
279    ///

281    ///     assert!(!turtle.trim().is_empty(), "Expected non-empty Turtle output");
282    ///
283    ///     // Very lightweight sanity check for Turtle syntax
284    ///     assert!(
285    ///         turtle.contains('.') || turtle.contains("@prefix"),
286    ///         "Output does not look like valid Turtle"
287    ///     );
288    ///
289    ///     Ok(())
290    /// }
291    /// ```
292    pub async fn query_construct_async(&self, query: &str, format: &QueryResultFormat) -> Result<String> {
293        // Select the appropriate client based on the requested format
294        let client = self.get_construct_client(format).await?;
295
296        make_sparql_query_construct_async(query, &client, &self.endpoint_iri, format).await
297    }

299    /// Retrieves or creates an HTTP client for the given `[[QueryResultFormat]]`
300    async fn get_construct_client(&self, format: &QueryResultFormat) -> Result<Arc<reqwest::Client>> {
301        {
302            let map = self.construct_clients.read().await;
303            if let Some(client) = map.get(format) {
304                return Ok(client.clone());
305            }
306        }
307        let mut headers = HeaderMap::new();
308        headers.insert(ACCEPT, HeaderValue::from_static(format.mime_type()));
309        headers.insert(USER_AGENT, HeaderValue::from_static("rudof"));
310
311        let client = reqwest::Client::builder().default_headers(headers).build()?;
312        let client = Arc::new(client);
313
314        let mut map = self.construct_clients.write().await;
315        Ok(map.entry(format.clone()).or_insert_with(|| client.clone()).clone())
316    }

360impl FromStr for SparqlEndpoint {
361    type Err = SparqlEndpointError;
362
363    /// Parses a SPARQL endpoint from a string.
364    ///
365    /// Supports two formats:
366    /// - IRI in angle brackets: `<https://example.org/sparql>`
367    /// - Predefined endpoint name: `wikidata`
368    ///
369    /// # Performance
370    ///
371    /// Uses a cached regex (via `once_cell::Lazy`) to avoid recompiling
372    /// the pattern on each call.
373    fn from_str(s: &str) -> std::result::Result<Self, Self::Err> {
374        use once_cell::sync::Lazy;
375        // Regex is compiled once and cached for all future calls
376        static RE_IRI: Lazy<Regex> = Lazy::new(|| Regex::new(r"<(.*)>").unwrap());
377
378        if let Some(iri_str) = RE_IRI.captures(s) {
379            // Parse IRI from angle brackets
380            let iri_s = IriS::from_str(&iri_str[1])?;
381            let client = Arc::new(sparql_client()?);
382
383            Ok(SparqlEndpoint {
384                endpoint_iri: iri_s,
385                prefixmap: Arc::new(PrefixMap::new()),
386                client,
387                construct_clients: Arc::new(RwLock::new(HashMap::new())),
388            })
389        } else {
390            // Try to match predefined endpoint names
391            match s.to_lowercase().as_str() {
392                "wikidata" => SparqlEndpoint::wikidata(),
393                name => Err(SparqlEndpointError::UnknownEndpointName { name: name.to_string() }),
394            }
395        }
396    }
397}

446impl AsyncRDF for SparqlEndpoint {

454    /// Retrieves all predicates for a given subject.
455    ///
456    /// Executes a SPARQL query: `SELECT ?pred WHERE { <subject> ?pred ?obj }`
457    ///
458    /// # Performance
459    ///
460    /// Pre-allocates the HashSet with capacity based on the number of results.
461    async fn get_predicates_subject(&self, subject: &OxSubject) -> Result<HashSet<OxNamedNode>> {
462        let query = format!(r#"select ?pred where {{ {subject} ?pred ?obj . }}"#);
463        let solutions = make_sparql_query_select_async(&query, &self.client, &self.endpoint_iri).await?;
464
465        let mut results = HashSet::with_capacity(solutions.len());
466        for solution in solutions {
467            let n = get_iri_solution(&solution, "pred")?;
468            results.insert(n);
469        }
470        Ok(results)
471    }
472
473    /// Retrieves all objects for a given subject-predicate pair.
474    ///
475    /// Executes a SPARQL query: `SELECT ?obj WHERE { <subject> <pred> ?obj }`
476    async fn get_objects_for_subject_predicate(
477        &self,
478        subject: &OxSubject,
479        pred: &OxNamedNode,
480    ) -> Result<HashSet<OxTerm>> {
481        let query = format!(r#"select ?obj where {{ {subject} {pred} ?obj . }}"#);
482        let solutions = make_sparql_query_select_async(&query, &self.client, &self.endpoint_iri).await?;
483
484        let mut results = HashSet::with_capacity(solutions.len());
485        for solution in solutions {
486            if let Some(obj) = solution.get("obj") {
487                results.insert(obj.clone());
488            }
489        }
490        Ok(results)
491    }
492
493    /// Retrieves all subjects for a given predicate-object pair.
494    ///
495    /// Executes a SPARQL query: `SELECT ?subj WHERE { ?subj <pred> <object> }`
496    async fn get_subjects_for_object_predicate(
497        &self,
498        object: &OxTerm,
499        pred: &OxNamedNode,
500    ) -> Result<HashSet<OxSubject>> {
501        let query = format!(r#"select ?subj where {{ ?subj {pred} {object} . }}"#);
502        let solutions = make_sparql_query_select_async(&query, &self.client, &self.endpoint_iri).await?;
503
504        let mut results = HashSet::with_capacity(solutions.len());
505        for solution in solutions {
506            if let Some(OxTerm::NamedNode(n)) = solution.get("subj") {
507                results.insert(OxSubject::NamedNode(n.clone()));
508            } else if let Some(OxTerm::BlankNode(bn)) = solution.get("subj") {
509                results.insert(OxSubject::BlankNode(bn.clone()));
510            }
511        }
512        Ok(results)
513    }
514}

516// NeighsRDF is only available on non-WASM platforms because it requires
517// synchronous iteration, which is not possible in WASM environments
518#[cfg(not(target_family = "wasm"))]
519impl NeighsRDF for SparqlEndpoint {
520    /// Returns an iterator over all triples in the endpoint.
521    ///
522    /// This is equivalent to `SELECT * WHERE { ?s ?p ?o }`.
523    ///
524    /// Note: This can be very expensive for large endpoints.
525    fn triples(&self) -> Result<impl Iterator<Item = Self::Triple>> {
526        self.triples_matching(&Any, &Any, &Any)
527    }
528
529    /// Returns an iterator over triples matching the given pattern.
530    ///
531    /// # Arguments
532    ///
533    /// * `subject` - Subject matcher (use `Any` to match all)
534    /// * `predicate` - Predicate matcher (use `Any` to match all)
535    /// * `object` - Object matcher (use `Any` to match all)
536    ///
537    /// # Examples
538    ///
539    /// ```no_run
540    /// use rudof_rdf::rdf_impl::SparqlEndpoint;
541    /// use rudof_rdf::rdf_core::{Any, NeighsRDF};
542    /// use oxrdf::NamedNode;
543    ///
544    /// fn main() -> Result<(), Box<dyn std::error::Error>> {
545    ///     let endpoint = SparqlEndpoint::wikidata()?;
546    ///
547    ///     let predicate = NamedNode::new("http://www.w3.org/1999/02/22-rdf-syntax-ns#type")?;
548    ///     let mut triples = endpoint.triples_matching(&Any, &predicate, &Any)?;
549    ///
550    ///     // Take the first triple, if any
551    ///     let first = triples.next();
552    ///
553    ///     assert!(first.is_some(), "Expected at least one triple from the endpoint");
554    ///
555    ///     Ok(())
556    /// }
557    /// ```
558    fn triples_matching<S, P, O>(
559        &self,
560        subject: &S,
561        predicate: &P,
562        object: &O,
563    ) -> Result<impl Iterator<Item = Self::Triple> + '_>
564    where
565        S: Matcher<Self::Subject>,
566        P: Matcher<Self::IRI>,
567        O: Matcher<Self::Term>,
568    {
569        // Build SPARQL query from matchers
570        let s_str = subject
571            .value()
572            .map(|s| s.to_string())
573            .unwrap_or_else(|| "?s".to_string());
574        let p_str = predicate
575            .value()
576            .map(|p| p.to_string())
577            .unwrap_or_else(|| "?p".to_string());
578        let o_str = object
579            .value()
580            .map(|o| o.to_string())
581            .unwrap_or_else(|| "?o".to_string());
582
583        let query = format!("SELECT ?s ?p ?o WHERE {{ {} {} {} }}", s_str, p_str, o_str);
584
585        let solutions = self.query_select(&query)?;
586
587        // Clone matcher values for use in the closure
588        let subject_val = subject.value().cloned();
589        let predicate_val = predicate.value().cloned();
590        let object_val = object.value().cloned();
591
592        // Build iterator that converts query solutions to triples
593        let triples = solutions.into_iter().filter_map(move |solution| {
594            // Use matched value if available, otherwise extract from solution
595            let subject_res: Self::Subject = match &subject_val {
596                Some(s) => s.clone(),
597                None => solution.find_solution(0).and_then(|s| s.clone().try_into().ok())?,
598            };
599
600            let predicate_res: Self::IRI = match &predicate_val {
601                Some(p) => p.clone(),
602                None => solution
603                    .find_solution(1)
604                    .and_then(|pred| pred.clone().try_into().ok())?,
605            };
606
607            let object_res = match &object_val {
608                Some(o) => o.clone(),
609                None => solution.find_solution(2)?.clone(),
610            };
611
612            Some(OxTriple::new(subject_res, predicate_res, object_res))
613        });
614
615        Ok(triples)
616    }
617}

619// QueryRDF is only available on non-WASM platforms.
620// On native platforms, these sync methods use tokio::runtime to run
621// the async implementations.
622#[cfg(not(target_family = "wasm"))]
623impl QueryRDF for SparqlEndpoint {
624    /// Executes a SPARQL CONSTRUCT query synchronously.
625    ///
626    /// This is a blocking wrapper around `query_construct_async`.
627    /// It creates a tokio runtime to run the async method.
628    ///
629    /// # Performance Note
630    ///
631    /// Creates a new runtime for each call. For better performance in
632    /// async contexts, use `query_construct_async` directly.
633    fn query_construct(&self, query: &str, format: &QueryResultFormat) -> Result<String> {
634        let runtime = tokio::runtime::Runtime::new().map_err(|e| SparqlEndpointError::ParsingBody {
635            body: format!("Failed to create runtime: {}", e),
636        })?;
637
638        runtime.block_on(self.query_construct_async(query, format))
639    }
640
641    /// Executes a SPARQL SELECT query synchronously.
642    ///
643    /// This is a blocking wrapper around `query_select_async`.
644    /// It creates a tokio runtime to run the async method.
645    ///
646    /// # Performance Note
647    ///
648    /// Creates a new runtime for each call. For better performance in
649    /// async contexts, use `query_select_async` directly.
650    fn query_select(&self, query: &str) -> Result<QuerySolutions<Self>> {
651        let runtime = tokio::runtime::Runtime::new().map_err(|e| SparqlEndpointError::ParsingBody {
652            body: format!("Failed to create runtime: {}", e),
653        })?;
654
655        runtime.block_on(self.query_select_async(query))
656    }
657
658    /// Executes a SPARQL ASK query synchronously.
659    ///
660    /// This is a blocking wrapper around `query_ask_async`.
661    /// It creates a tokio runtime to run the async method.
662    ///
663    /// # Performance Note
664    ///
665    /// Creates a new runtime for each call. For better performance in
666    /// async contexts, use `query_ask_async` directly.
667    fn query_ask(&self, query: &str) -> Result<bool> {
668        let runtime = tokio::runtime::Runtime::new().map_err(|e| SparqlEndpointError::ParsingBody {
669            body: format!("Failed to create runtime: {}", e),
670        })?;
671
672        runtime.block_on(self.query_ask_async(query))
673    }
674}

677///
678/// # Performance
679///

682fn cnv_query_solution(qs: &OxQuerySolution) -> QuerySolution<SparqlEndpoint> {

688/// Creates an HTTP client configured for SPARQL SELECT queries.
689///
690/// Sets the Accept header to `application/sparql-results+json` and
691/// includes a custom User-Agent.
692///
693/// # Errors
694///
695/// Returns an error if the client builder fails (e.g., TLS initialization fails).
696fn sparql_client() -> Result<reqwest::Client> {
697    use reqwest::header::{self, ACCEPT, USER_AGENT};
698
699    let mut headers = header::HeaderMap::new();
700    headers.insert(
701        ACCEPT,
702        header::HeaderValue::from_static("application/sparql-results+json"),
703    );
704    headers.insert(USER_AGENT, header::HeaderValue::from_static("rudof"));
705
706    let client = reqwest::Client::builder().default_headers(headers).build()?;
707    Ok(client)
708}

800/// Parses SPARQL ASK query JSON results.
801///
802/// ASK queries return JSON in the format: `{"head": {}, "boolean": true}`
803///
804/// # Arguments
805///
806/// * `body` - The JSON response body as a string
807///
808/// # Returns
809///
810/// The boolean value from the response.
811///
812/// # Errors
813///
814/// Returns an error if:
815/// - The JSON cannot be parsed
816/// - The response is not a boolean result
817fn parse_sparql_ask_results(body: &str) -> Result<bool> {
818    let json_parser = QueryResultsParser::from_format(QueryResultsFormat::Json);
819
820    match json_parser.for_reader(body.as_bytes())? {
821        ReaderQueryResultsParserOutput::Boolean(b) => Ok(b),
822        _ => Err(SparqlEndpointError::ParsingBody {
823            body: format!("Expected boolean ASK result, got: {}", body),
824        }),
825    }
826}

828/// Parses SPARQL JSON results into a vector of query solutions.
829///
830/// # Arguments
831///
832/// * `body` - The JSON response body as a string
833///
834/// # Returns
835///
836/// A vector of query solutions.
837///
838/// # Errors
839///
840/// Returns an error if:
841/// - The JSON cannot be parsed
842/// - The JSON is not a valid SPARQL results format
843/// - Individual solutions cannot be parsed
844fn parse_sparql_json_results(body: &str) -> Result<Vec<OxQuerySolution>> {
845    let json_parser = QueryResultsParser::from_format(QueryResultsFormat::Json);
846
847    if let ReaderQueryResultsParserOutput::Solutions(solutions) = json_parser.for_reader(body.as_bytes())? {
848        // Collect all solutions, propagating any parsing errors
849        solutions
850            .collect::<std::result::Result<Vec<_>, _>>()
851            .map_err(|e| SparqlEndpointError::ParsingBody {
852                body: format!("Error parsing solution: {}", e),
853            })
854    } else {
855        Err(SparqlEndpointError::ParsingBody { body: body.to_string() })
856    }
857}

875/// Extracts a named IRI from a query solution.
876///
877/// # Arguments
878///
879/// * `solution` - The query solution
880/// * `name` - The variable name to extract
881///
882/// # Returns
883///
884/// The named node if found and is an IRI.
885///
886/// # Errors
887///
888/// Returns an error if:
889/// - The variable is not found in the solution
890/// - The value is not a named node (IRI)
891fn get_iri_solution(solution: &OxQuerySolution, name: &str) -> Result<OxNamedNode> {
892    solution
893        .get(name)
894        .ok_or_else(|| SparqlEndpointError::NotFoundInSolution {
895            value: name.to_string(),
896            solution: format!("{solution:?}"),
897        })
898        .and_then(|v| match v {
899            OxTerm::NamedNode(n) => Ok(n.clone()),
900            _ => Err(SparqlEndpointError::SPARQLSolutionErrorNoIRI { value: v.clone() }),
901        })
902}
