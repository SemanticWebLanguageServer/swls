37/// An RDF graph stored entirely in memory.
38///
39/// The graph is backed by [`oxrdf::Graph`] and enriched with prefix handling,
40/// base IRI support, blank node generation, and optional SPARQL querying via
41/// an Oxigraph [`Store`].
42#[derive(Default, Clone)]
43pub struct InMemoryGraph {
44    /// Optional focus term used by [`FocusRDF`] operations.
45    focus: Option<OxTerm>,
46
47    /// Underlying RDF graph.
48    graph: Arc<Graph>,
49
50    /// Prefix map used for CURIE resolution and qualification.
51    pm: PrefixMap,
52
53    /// Optional base IRI for resolving relative IRIs.
54    base: Option<IriS>,
55
56    /// Counter used to generate unique blank node identifiers.
57    bnode_counter: usize,
58
59    /// Optional Oxigraph store used for SPARQL evaluation.
60    store: Option<Store>,
61}

108    /// Merges RDF data from a reader into the current graph.
109    ///
110    /// The parsing behavior depends on [`RDFFormat`] and [`ReaderMode`].
111    /// Prefixes and base IRI are merged when available.
112    ///
113    /// # Parameters
114    ///
115    /// * `reader` - Input stream containing RDF data
116    /// * `source_name` - Name used for error reporting
117    /// * `format` - RDF serialization format
118    /// * `base` - Optional base IRI for resolving relative IRIs
119    /// * `reader_mode` - Controls error handling (strict or lax)
120    ///
121    /// # Errors
122    ///
123    /// Returns an error if parsing fails in strict mode or if I/O errors occur.
124    pub fn merge_from_reader<R: io::Read>(
125        &mut self,
126        reader: &mut R,
127        source_name: &str,
128        format: &RDFFormat,
129        base: Option<&str>,
130        reader_mode: &ReaderMode,
131    ) -> Result<(), InMemoryGraphError> {
132        match format {
133            RDFFormat::Turtle => {
134                self.parse_turtle(reader, source_name, base, reader_mode)?;
135            },
136            RDFFormat::NTriples => {
137                self.parse_ntriples(reader, reader_mode)?;
138            },
139            RDFFormat::Rdfxml => {
140                self.parse_rdfxml(reader, reader_mode)?;
141            },
142            RDFFormat::TriG => {
143                todo!();
144            },
145            RDFFormat::N3 => {
146                todo!();
147            },
148            RDFFormat::NQuads => {
149                self.parse_nquads(reader, reader_mode)?;
150            },
151            RDFFormat::JsonLd => {
152                self.parse_jsonld(reader, reader_mode)?;
153            },
154        }
155        Ok(())
156    }

158    /// Parses Turtle data and merges it into the graph.
159    ///
160    /// # Parameters
161    ///
162    /// * `reader` - Input stream containing Turtle data
163    /// * `source_name` - Name used for error reporting
164    /// * `base` - Optional base IRI for resolving relative IRIs
165    /// * `reader_mode` - Controls error handling (strict or lax)
166    ///
167    /// # Errors
168    ///
169    /// Returns an error if parsing fails in strict mode.
170    fn parse_turtle<R: io::Read>(
171        &mut self,
172        reader: &mut R,
173        source_name: &str,
174        base: Option<&str>,
175        reader_mode: &ReaderMode,
176    ) -> Result<(), InMemoryGraphError> {
177        let turtle_parser = match base {
178            None => TurtleParser::new(),
179            Some(iri) => TurtleParser::new().with_base_iri(iri)?,
180        };
181
182        let mut buffer = Vec::new();
183        reader.read_to_end(&mut buffer)?;
184        let reader1 = Cursor::new(&buffer);
185        let mut turtle_reader = turtle_parser.for_reader(reader1);
186
187        for triple_result in turtle_reader.by_ref() {
188            let triple =
189                match handle_parse_error(triple_result, reader_mode, |e| InMemoryGraphError::TurtleParseError {
190                    source_name: source_name.to_string(),
191                    error: e,
192                })? {
193                    Some(t) => t,
194                    None => continue,
195                };
196            Arc::make_mut(&mut self.graph).insert(triple.as_ref());
197        }
198
199        let prefixes: HashMap<&str, &str> = turtle_reader.prefixes().collect();
200        self.base = match (&self.base, base) {
201            (None, None) => None,
202            (Some(b), None) => Some(b.clone()),
203            (_, Some(b)) => Some(IriS::new_unchecked(b)),
204        };
205        let pm = PrefixMap::from_hashmap(prefixes)?;
206        self.merge_prefixes(pm)?;
207
208        Ok(())
209    }

569/// Implementation of the core `Rdf` trait.
570///
571/// This implementation provides the fundamental RDF operations including type definitions,
572/// prefix resolution, and term qualification (converting full IRIs to prefixed names).
573impl Rdf for InMemoryGraph {
574    type IRI = OxNamedNode;
575    type BNode = OxBlankNode;
576    type Literal = OxLiteral;
577    type Subject = OxSubject;
578    type Term = OxTerm;
579    type Triple = OxTriple;

582    /// Resolves a prefix and local name to a full IRI.
583    ///
584    /// # Parameters
585    ///
586    /// * `prefix` - The namespace prefix (e.g., "foaf")
587    /// * `local` - The local name (e.g., "Person")
588    ///
589    /// # Returns
590    ///
591    /// The full IRI (e.g., "http://xmlns.com/foaf/0.1/Person")
592    ///
593    /// # Errors
594    ///
595    /// Returns an error if the prefix is not defined in the prefix map.
596    fn resolve_prefix_local(&self, prefix: &str, local: &str) -> Result<IriS, PrefixMapError> {
597        let iri = self.pm.resolve_prefix_local(prefix, local)?;
598        Ok(iri)
599    }

601    /// Converts a full IRI to a qualified (prefixed) name if possible.
602    ///
603    /// If the IRI matches a known namespace prefix, it returns a shortened form
604    /// (e.g., "foaf:Person"). Otherwise, it returns the full IRI.
605    ///
606    /// # Parameters
607    ///
608    /// * `node` - The named node to qualify
609    ///
610    /// # Returns
611    ///
612    /// A string representation, either prefixed or full IRI.
613    fn qualify_iri(&self, node: &Self::IRI) -> String {
614        let iri = IriS::from_str(node.as_str()).expect("OxNamedNode should always contain valid IRI");
615        self.pm.qualify(&iri)
616    }
617
618    /// Converts a subject (named node or blank node) to a qualified string.
619    ///
620    /// Named nodes are qualified using the prefix map. Blank nodes are formatted
621    /// with color for display.
622    ///
623    /// # Parameters
624    ///
625    /// * `subj` - The subject to qualify
626    ///
627    /// # Returns
628    ///
629    /// A string representation of the subject.
630    fn qualify_subject(&self, subj: &OxSubject) -> String {
631        match subj {
632            OxSubject::BlankNode(bn) => self.show_blanknode(bn),
633            OxSubject::NamedNode(n) => self.qualify_iri(n),
634        }
635    }
636
637    /// Converts an RDF term (named node, blank node, or literal) to a qualified string.
638    ///
639    /// Different term types are formatted differently:
640    /// - Named nodes: qualified using prefix map
641    /// - Blank nodes: formatted with green color
642    /// - Literals: formatted with red color
643    /// - RDF-star triples: not yet supported
644    ///
645    /// # Parameters
646    ///
647    /// * `term` - The term to qualify
648    ///
649    /// # Returns
650    ///
651    /// A string representation of the term.
652    fn qualify_term(&self, term: &OxTerm) -> String {
653        match term {
654            OxTerm::BlankNode(bn) => self.show_blanknode(bn),
655            OxTerm::Literal(lit) => self.show_literal(lit),
656            OxTerm::NamedNode(n) => self.qualify_iri(n),
657            OxTerm::Triple(_) => unimplemented!("RDF-star triples not yet supported"),
658        }
659    }

664    ///
665    /// `Some(&PrefixMap)` containing all defined namespace prefixes.
666    fn prefixmap(&self) -> Option<PrefixMap> {
667        Some(self.pm.clone())
668    }

691    ///
692    /// * `subject` - Pattern matcher for the subject
693    /// * `predicate` - Pattern matcher for the predicate
694    /// * `object` - Pattern matcher for the object
695    ///
696    /// # Returns
697    ///
698    /// An iterator that yields triples matching all three patterns.
699    fn triples_matching<S, P, O>(
700        &self,
701        subject: &S,
702        predicate: &P,
703        object: &O,
704    ) -> Result<impl Iterator<Item = Self::Triple> + '_, Self::Err>
705    where
706        S: Matcher<Self::Subject>,
707        P: Matcher<Self::IRI>,
708        O: Matcher<Self::Term>,
709    {
710        // Dispatch to the tightest available index to avoid full scans.
711        let result: Box<dyn Iterator<Item = OxTriple> + '_> = match (subject.value(), predicate.value(), object.value())
712        {
713            (Some(s), Some(p), Some(o)) => {
714                let found = self.graph.contains(TripleRef::new(
715                    OxSubjectRef::from(s),
716                    NamedNodeRef::from(p),
717                    TermRef::from(o),
718                ));
719                let s = s.clone();
720                let p = p.clone();
721                let o = o.clone();
722                Box::new(found.then(|| OxTriple::new(s, p, o)).into_iter())
723            },
724            (Some(s), Some(p), None) => {
725                let s_owned = s.clone();
726                let p_owned = p.clone();
727                Box::new(
728                    self.graph
729                        .objects_for_subject_predicate(OxSubjectRef::from(s), NamedNodeRef::from(p))
730                        .map(move |o| OxTriple::new(s_owned.clone(), p_owned.clone(), o.into_owned())),
731                )
732            },
733            (Some(s), None, Some(o)) => {
734                let o_owned = o.clone();
735                Box::new(
736                    self.graph
737                        .triples_for_subject(OxSubjectRef::from(s))
738                        .map(TripleRef::into_owned)
739                        .filter(move |t| t.object == o_owned),
740                )
741            },
742            (Some(s), None, None) => Box::new(
743                self.graph
744                    .triples_for_subject(OxSubjectRef::from(s))
745                    .map(TripleRef::into_owned),
746            ),
747            (None, Some(p), Some(o)) => {
748                let p_owned = p.clone();
749                let o_owned = o.clone();
750                Box::new(
751                    self.graph
752                        .subjects_for_predicate_object(NamedNodeRef::from(p), TermRef::from(o))
753                        .map(move |s| OxTriple::new(s.into_owned(), p_owned.clone(), o_owned.clone())),
754                )
755            },
756            (None, Some(p), None) => Box::new(
757                self.graph
758                    .triples_for_predicate(NamedNodeRef::from(p))
759                    .map(TripleRef::into_owned),
760            ),
761            (None, None, Some(o)) => Box::new(
762                self.graph
763                    .triples_for_object(TermRef::from(o))
764                    .map(TripleRef::into_owned),
765            ),
766            (None, None, None) => Box::new(self.graph.iter().map(TripleRef::into_owned)),
767        };
768
769        Ok(result)
770    }
771}

1031        writer.finish()?;
1032        Ok(())

1085/// Helper function to handle parse errors consistently.
1086///
1087/// This function implements a consistent error handling strategy across all parsers.
1088/// In strict mode, errors are propagated. In lax mode, errors are logged and parsing continues.
1089///
1090/// # Parameters
1091///
1092/// * `result` - The parse result to handle
1093/// * `reader_mode` - Controls error handling behavior
1094/// * `error_constructor` - Function to construct an appropriate error type
1095///
1096/// # Returns
1097///
1098/// * `Ok(Some(value))` - Parsing succeeded
1099/// * `Ok(None)` - Parsing failed in lax mode (skip this item)
1100/// * `Err(error)` - Parsing failed in strict mode
1101fn handle_parse_error<T, E: std::fmt::Display>(
1102    result: Result<T, E>,
1103    reader_mode: &ReaderMode,
1104    error_constructor: impl FnOnce(String) -> InMemoryGraphError,
1105) -> Result<Option<T>, InMemoryGraphError> {
1106    match result {
1107        Ok(val) => Ok(Some(val)),
1108        Err(e) => {
1109            if reader_mode.is_strict() {
1110                Err(error_constructor(e.to_string()))
1111            } else {
1112                Ok(None)
1113            }
1114        },
1115    }
1116}

1277/// Converts a single Oxigraph query solution to internal representation.
1278///

1289    for v in qs.variables() {
