7/// Represents an RDF resource that is either an IRI or a blank node.
8///
9/// # Variants
10/// - **BlankNode**: An anonymous resource without a global identifier, represented
11///   by a local label (e.g., "b0", "genid1")
12/// - **Iri**: A globally identified resource using an Internationalized Resource
13///   Identifier
14#[derive(Debug, Clone, Hash, PartialEq, Eq, Serialize, Deserialize, PartialOrd, Ord)]
15pub enum IriOrBlankNode {
16    BlankNode(String),
17    Iri(IriS),
18}
19
20impl IriOrBlankNode {
21    /// Returns the length (in bytes) of this resource's string representation.
22    ///
23    /// - For blank nodes: the length of the identifier string
24    /// - For IRIs: the length of the full IRI string
25    pub fn length(&self) -> usize {
26        match self {
27            IriOrBlankNode::BlankNode(label) => label.len(),
28            IriOrBlankNode::Iri(iri) => iri.as_str().len(),
29        }
30    }
31
32    /// Creates an IRI resource from an `IriS` reference.
33    ///
34    /// # Parameters
35    /// - `iri`: Reference to the IRI to wrap
36    pub fn iri(iri: &IriS) -> IriOrBlankNode {
37        IriOrBlankNode::Iri(iri.clone())
38    }
39
40    /// Formats this resource using qualified names (prefixes) where possible.
41    ///
42    /// This method produces a compact representation by replacing full IRIs
43    /// with prefixed names when a matching prefix exists in the provided map.
44    ///
45    /// - Blank nodes: formatted as "_:<id>"
46    /// - IRIs: qualified using the prefix map (e.g., "ex:resource" instead of
47    ///   "http://example.org/resource")
48    ///
49    /// # Parameters
50    /// - `prefixmap`: A prefix map containing IRI-to-prefix mappings
51    pub fn show_qualified(&self, prefixmap: &prefixmap::PrefixMap) -> String {
52        match self {
53            IriOrBlankNode::BlankNode(bnode) => format!("_:{bnode}"),
54            IriOrBlankNode::Iri(iri) => prefixmap.qualify(iri),
55        }
56    }
57}

63impl Display for IriOrBlankNode {
64    /// Formats the resource for display (human-readable output).
65    ///
66    /// - Blank nodes: displayed as the identifier string without "_:" prefix
67    /// - IRIs: displayed as the full IRI string
68    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
69        match self {
70            IriOrBlankNode::BlankNode(b) => write!(f, "{b}"),
71            IriOrBlankNode::Iri(iri_s) => write!(f, "{iri_s}"),
72        }
73    }
74}

80/// Converts an `IriOrBlankNode` into an `oxrdf::NamedOrBlankNode`.
81///
82/// This enables interoperability with the `oxrdf` library, allowing seamless
83/// conversion from the custom representation to oxrdf's subject type.
84impl From<IriOrBlankNode> for oxrdf::NamedOrBlankNode {
85    fn from(value: IriOrBlankNode) -> Self {
86        match value {
87            IriOrBlankNode::Iri(iri) => oxrdf::NamedNode::new_unchecked(iri.as_str()).into(),
88            IriOrBlankNode::BlankNode(bnode) => oxrdf::BlankNode::new_unchecked(bnode).into(),
89        }
90    }
91}

93/// Converts an `oxrdf::NamedOrBlankNode` into an `IriOrBlankNode`.
94///
95/// This enables interoperability with the `oxrdf` library, allowing conversion
96/// from oxrdf's subject type into the custom representation.
97impl From<oxrdf::NamedOrBlankNode> for IriOrBlankNode {
98    fn from(value: oxrdf::NamedOrBlankNode) -> Self {
99        match value {
100            oxrdf::NamedOrBlankNode::NamedNode(iri) => IriOrBlankNode::Iri(iri.into()),
101            oxrdf::NamedOrBlankNode::BlankNode(bnode) => IriOrBlankNode::BlankNode(bnode.into_string()),
102        }
103    }
104}

106/// Converts an `IriOrBlankNode` into an `oxrdf::Term`.
107///
108/// This conversion allows using `IriOrBlankNode` values in contexts where
109/// oxrdf expects a full term (which can be an IRI, blank node, literal, or triple).
110impl From<IriOrBlankNode> for oxrdf::Term {
111    fn from(value: IriOrBlankNode) -> Self {
112        match value {
113            IriOrBlankNode::Iri(iri) => oxrdf::Term::NamedNode(oxrdf::NamedNode::new_unchecked(iri.as_str())),
114            IriOrBlankNode::BlankNode(bnode) => oxrdf::Term::BlankNode(oxrdf::BlankNode::new_unchecked(bnode)),
115        }
116    }
117}

119/// Attempts to convert an `oxrdf::Term` into an `IriOrBlankNode`.
120///
121/// This conversion succeeds only when the term is a named node or blank node.
122/// It fails if the term is a literal or RDF-star triple.
123///
124/// # Errors
125///
126/// Returns an error if the term is a literal or triple, as these cannot be
127/// represented as `IriOrBlankNode`.
128impl TryFrom<oxrdf::Term> for IriOrBlankNode {
129    type Error = RDFError;
130
131    fn try_from(value: oxrdf::Term) -> Result<Self, Self::Error> {
132        match value {
133            oxrdf::Term::NamedNode(iri) => Ok(IriOrBlankNode::Iri(iri.into())),
134            oxrdf::Term::BlankNode(bnode) => Ok(IriOrBlankNode::BlankNode(bnode.into_string())),
135            oxrdf::Term::Literal(lit) => Err(RDFError::ExpectedIriOrBlankNodeFoundLiteral {
136                literal: lit.to_string(),
137            }),
138            oxrdf::Term::Triple(triple) => Err(RDFError::ExpectedIriOrBlankNodeFoundTriple {
139                subject: triple.subject.to_string(),
140                predicate: triple.predicate.to_string(),
141                object: triple.object.to_string(),
142            }),
143        }
144    }
145}

147/// Attempts to convert an `Object` into an `IriOrBlankNode`.
148///
149/// This conversion succeeds only when the object is an IRI or blank node.
150/// It fails with an error if the object is a literal or triple, as these
151/// cannot be represented as resources.
152///
153/// # Errors
154///
155/// - `RDFError::ExpectedIriOrBlankNodeFoundLiteral`: When attempting to convert
156///   a literal object. Literals cannot be subjects in standard RDF.
157/// - `RDFError::ExpectedIriOrBlankNodeFoundTriple`: When attempting to convert
158///   an RDF-star triple object in a context where quoted triples are not supported.
159impl TryFrom<Object> for IriOrBlankNode {
160    type Error = RDFError;
161
162    fn try_from(value: Object) -> Result<Self, Self::Error> {
163        match value {
164            Object::Iri(iri) => Ok(IriOrBlankNode::Iri(iri)),
165            Object::BlankNode(b) => Ok(IriOrBlankNode::BlankNode(b)),
166            Object::Literal(l) => Err(RDFError::ExpectedIriOrBlankNodeFoundLiteral { literal: l.to_string() }),
167            Object::Triple {
168                subject,
169                predicate,
170                object,
171            } => Err(RDFError::ExpectedIriOrBlankNodeFoundTriple {
172                subject: subject.to_string(),
173                predicate: predicate.to_string(),
174                object: object.to_string(),
175            }),
176        }
177    }
