11pub enum NodeExpr<RDF: Rdf> {
12    // Constants
13    Iri(IriRef),
14    Literal(ConcreteLiteral),
15
16    // Blank Node
17    NamedParameterFn(HashMap<IriRef, NodeExpr<RDF>>),
18    ListParameterFn {
19        property: IriRef,
20        args: Vec<NodeExpr<RDF>>,
21    },
22
23    // Basic expressions
24    Empty,
25    Var(String),
26    List(Vec<Object>),
27    PathValues {
28        path: SHACLPath,
29        focus_node: Option<Box<NodeExpr<RDF>>>,
30    },
31    Exists(Box<NodeExpr<RDF>>),
32    IfExpression {
33        if_condition: Box<NodeExpr<RDF>>,
34        then: Box<NodeExpr<RDF>>,
35        else_expression: Box<NodeExpr<RDF>>,
36    },
37
38    // List expressions
39    Distinct(Box<NodeExpr<RDF>>),
40    Intersection(Vec<NodeExpr<RDF>>),
41    Concat(Vec<NodeExpr<RDF>>),
42    Remove {
43        remove: Box<NodeExpr<RDF>>,
44        nodes: Box<NodeExpr<RDF>>,
45    },
46    Filter {
47        filter_shape: Shape<RDF>,
48        nodes: Box<NodeExpr<RDF>>,
49    },
50    Limit {
51        limit: usize,
52        nodes: Box<NodeExpr<RDF>>,
53    },
54    Offset {
55        offset: usize,
56        nodes: Box<NodeExpr<RDF>>,
57    },
58    // OrderBy, // TODO - Not yet defined in the draft
59
60    // Advance sequence operations
61    FlatMap {
62        flat_map: Box<NodeExpr<RDF>>,
63        nodes: Box<NodeExpr<RDF>>,
64    },
65    FindFirst {
66        find_first: Shape<RDF>,
67        nodes: Box<NodeExpr<RDF>>,
68    },
69    MatchAll {
70        match_all: Shape<RDF>,
71        nodes: Box<NodeExpr<RDF>>,
72    },
73
74    // Aggregation expressions
75    Count(Box<NodeExpr<RDF>>),
76    Min(Box<NodeExpr<RDF>>),
77    Max(Box<NodeExpr<RDF>>),
78    Sum(Box<NodeExpr<RDF>>), // The draft mentions that maybe is removed
79
80    // Miscellaneous expressions
81    InstancesOf(IriRef),
82    NodesMatching(IriOrBlankNode),
83}

85impl<RDF: Rdf> Display for NodeExpr<RDF> {
86    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
87        match self {
88            NodeExpr::Iri(i) => write!(f, "iri({i})"),
89            NodeExpr::Literal(l) => write!(f, "literal({l})"),
90            NodeExpr::NamedParameterFn(map) => {
91                let params = map
92                    .iter()
93                    .map(|(k, v)| format!("{}: {}", k, v))
94                    .collect::<Vec<_>>()
95                    .join(", ");
96                write!(f, "namedParameterFn({params})")
97            },
98            NodeExpr::ListParameterFn { args, property } => {
99                let args_str = args.iter().map(|arg| arg.to_string()).collect::<Vec<_>>().join(", ");
100                write!(f, "listParameterFn({property}, {args_str})")
101            },
102            NodeExpr::Empty => write!(f, "empty"),
103            NodeExpr::Var(v) => write!(f, "var({v})"),
104            NodeExpr::List(l) => write!(
105                f,
106                "list({})",
107                l.iter().map(|o| o.to_string()).collect::<Vec<_>>().join(", ")
108            ),
109            NodeExpr::PathValues { focus_node, path } => {
110                let fnode = match focus_node {
111                    None => "".to_string(),
112                    Some(n) => n.to_string(),
113                };
114                write!(f, "pathValues({path}, {fnode})")
115            },
116            NodeExpr::Exists(e) => write!(f, "exists({e})"),
117            NodeExpr::IfExpression {
118                if_condition,
119                then,
120                else_expression,
121            } => {
122                write!(f, "if({if_condition} then {then} else {else_expression})")
123            },
124            NodeExpr::Distinct(d) => write!(f, "distinct({d})"),
125            NodeExpr::Intersection(i) => write!(
126                f,
127                "intersection({})",
128                i.iter().map(|e| e.to_string()).collect::<Vec<_>>().join(", ")
129            ),
130            NodeExpr::Concat(c) => write!(
131                f,
132                "concat({})",
133                c.iter().map(|e| e.to_string()).collect::<Vec<_>>().join(", ")
134            ),
135            NodeExpr::Remove { remove, nodes } => write!(f, "remove({remove} in {nodes})"),
136            NodeExpr::Filter { filter_shape, nodes } => write!(f, "filter({filter_shape} in {nodes})"),
137            NodeExpr::Limit { limit, nodes } => writeln!(f, "limit({limit} in {nodes})"),
138            NodeExpr::Offset { offset, nodes } => writeln!(f, "offset({offset} in {nodes})"),
139            NodeExpr::FlatMap { flat_map, nodes } => write!(f, "flatMap({flat_map} in {nodes}"),
140            NodeExpr::FindFirst { find_first, nodes } => write!(f, "findFirst({find_first} in {nodes}"),
141            NodeExpr::MatchAll { match_all, nodes } => write!(f, "matchAll({match_all} in {nodes}"),
142            NodeExpr::Count(c) => write!(f, "count({c})"),
143            NodeExpr::Min(m) => write!(f, "min({m})"),
144            NodeExpr::Max(m) => write!(f, "max({m}"),
145            NodeExpr::Sum(s) => write!(f, "sum({s})"),
146            NodeExpr::InstancesOf(i) => write!(f, "instancesOf({i})"),
147            NodeExpr::NodesMatching(n) => write!(f, "nodesMatching({n})"),
148        }
149    }
150}

152impl<RDF: Rdf> Clone for NodeExpr<RDF> {

166                path: path.clone(),

168            },
169            NodeExpr::Exists(e) => NodeExpr::Exists(e.clone()),
170            NodeExpr::IfExpression {
171                if_condition,
172                then,
173                else_expression,
174            } => NodeExpr::IfExpression {
175                if_condition: if_condition.clone(),
176                then: then.clone(),
177                else_expression: else_expression.clone(),
178            },
179            NodeExpr::Distinct(d) => NodeExpr::Distinct(d.clone()),
180            NodeExpr::Intersection(i) => NodeExpr::Intersection(i.clone()),

182            NodeExpr::Remove { remove, nodes } => NodeExpr::Remove {
183                remove: remove.clone(),

186            NodeExpr::Filter { filter_shape, nodes } => NodeExpr::Filter {

220impl<RDF: Rdf> PartialEq for NodeExpr<RDF> {
221    fn eq(&self, other: &Self) -> bool {
222        match (self, other) {
223            (NodeExpr::Iri(l), NodeExpr::Iri(r)) => l == r,
224            (NodeExpr::Literal(l), NodeExpr::Literal(r)) => l == r,
225            (NodeExpr::NamedParameterFn(l), NodeExpr::NamedParameterFn(r)) => l == r,
226            (
227                NodeExpr::ListParameterFn { args: al, property: pl },
228                NodeExpr::ListParameterFn { args: ar, property: pr },
229            ) => al == ar && pl == pr,
230            (NodeExpr::Empty, NodeExpr::Empty) => true,
231            (NodeExpr::Var(l), NodeExpr::Var(r)) => l == r,
232            (NodeExpr::List(l), NodeExpr::List(r)) => l == r,
233            (
234                NodeExpr::PathValues {
235                    focus_node: fl,
236                    path: pl,
237                },
238                NodeExpr::PathValues {
239                    focus_node: fr,
240                    path: pr,
241                },
242            ) => pl == pr && fl == fr,
243            (NodeExpr::Exists(l), NodeExpr::Exists(r)) => l == r,
244            (
245                NodeExpr::IfExpression {
246                    if_condition: il,
247                    then: tl,
248                    else_expression: el,
249                },
250                NodeExpr::IfExpression {
251                    if_condition: ir,
252                    then: tr,
253                    else_expression: er,
254                },
255            ) => il == ir && tl == tr && el == er,
256            (NodeExpr::Distinct(l), NodeExpr::Distinct(r)) => l == r,
257            (NodeExpr::Intersection(l), NodeExpr::Intersection(r)) => l == r,
258            (NodeExpr::Concat(l), NodeExpr::Concat(r)) => l == r,
259            (NodeExpr::Remove { remove: rl, nodes: nl }, NodeExpr::Remove { remove: rr, nodes: nr }) => {
260                rl == rr && nl == nr
261            },
262            (
263                NodeExpr::Filter {
264                    filter_shape: fl,
265                    nodes: nl,
266                },
267                NodeExpr::Filter {
268                    filter_shape: fr,
269                    nodes: nr,
270                },
271            ) => fl == fr && nl == nr,
272            (NodeExpr::Limit { limit: ll, nodes: nl }, NodeExpr::Limit { limit: lr, nodes: nr }) => {
273                ll == lr && nl == nr
274            },
275            (NodeExpr::Offset { offset: ol, nodes: nl }, NodeExpr::Offset { offset: or, nodes: nr }) => {
276                ol == or && nl == nr
277            },
278            (
279                NodeExpr::FlatMap {
280                    flat_map: fl,
281                    nodes: nl,
282                },
283                NodeExpr::FlatMap {
284                    flat_map: fr,
285                    nodes: nr,
286                },
287            ) => fl == fr && nl == nr,
288            (
289                NodeExpr::FindFirst {
290                    find_first: fl,
291                    nodes: nl,
292                },
293                NodeExpr::FindFirst {
294                    find_first: fr,
295                    nodes: nr,
296                },
297            ) => fl == fr && nl == nr,
298            (
299                NodeExpr::MatchAll {
300                    match_all: ml,
301                    nodes: nl,
302                },
303                NodeExpr::MatchAll {
304                    match_all: mr,
305                    nodes: nr,
306                },
307            ) => ml == mr && nl == nr,
308            (NodeExpr::Count(l), NodeExpr::Count(r)) => l == r,
309            (NodeExpr::Min(l), NodeExpr::Min(r)) => l == r,
310            (NodeExpr::Max(l), NodeExpr::Max(r)) => l == r,
311            (NodeExpr::Sum(l), NodeExpr::Sum(r)) => l == r,
312            (NodeExpr::InstancesOf(l), NodeExpr::InstancesOf(r)) => l == r,
313            (NodeExpr::NodesMatching(l), NodeExpr::NodesMatching(r)) => l == r,
314            _ => false,
315        }
316    }
