29    let base = j(Token::SparqlBase)
30        .map_with_span(spanned)
31        .then(named_node().map_with_span(spanned))
32        .map(|(token, iri)| Prologue::Base { token, iri });
33
34    let prefix = j(Token::SparqlPrefix).map_with_span(|_, s| s)
35        .then(select! { |span| PToken(Token::PNameLN(x, _), _) => Spanned(x.unwrap_or_default(), span)})
36        .then(named_node()
37        .map_with_span(spanned))
38        .map(|((span, prefix), value)| {
39            Prologue::Prefix(TurtlePrefix {
40                span,
41                prefix,
42                value,
43            })
44        });
45
46    base.or(prefix)
47}

78fn select_clause() -> impl Parser<PToken, SelectClause, Error = Simple<PToken>> + Clone {
79    let star = j(Token::SparqlExpr(SparqlExpr::Times))
80        .to(Solution::All)
81        .map_with_span(spanned)
82        .map(|x| vec![x]);
83
84    let others = bind()
85        .delimited_by(j(Token::CurlOpen), j(Token::CurlClose))
86        .map(Solution::VarAs)
87        .or(variable().map(Solution::Var))
88        .map_with_span(spanned)
89        .repeated();
90
91    // This might be different to support ask and describe
92
93    sparql_kwd(SparqlKeyword::Select)
94        .then(
95            sparql_kwd(SparqlKeyword::Distinct)
96                .or(sparql_kwd(SparqlKeyword::Reduced))
97                .or_not(),
98        )
99        .then(star.or(others))
100        .map(|((kwd, modifier), solutions)| SelectClause {
101            kwd,
102            modifier,
103            solutions,
104        })
105}

108fn sub_select<'a>(
109    ctx: Ctx<'a>,
110) -> impl Parser<PToken, SubSelect, Error = Simple<PToken>> + Clone + use<'a> {
111    recursive(|sub_select| {

115            .then(modi)

125fn group_graph_pattern_sub<
126    'a,
127    T: Parser<PToken, GroupGraphPattern, Error = Simple<PToken>> + Clone,
128>(
129    ggp: T,
130    ctx: Ctx<'a>,
131) -> impl Parser<PToken, GroupGraphPatternSub, Error = Simple<PToken>> + Clone + use<'a, T> {
132    let next_check = not(Token::CurlClose).rewind();
133
134    let trip = triple(ctx)
135        .map_with_span(spanned)
136        .map(GroupGraphPatternSub::Triple)
137        .labelled("triple");
138
139    let kwd = sparql_kwd(SparqlKeyword::Minus)
140        .or(sparql_kwd(SparqlKeyword::Optional))
141        .clone()
142        .then_ignore(j(Token::CurlOpen).rewind())
143        .then(ggp.clone().map_with_span(spanned))
144        .map(|(kwd, ggp)| GroupGraphPatternSub::Kwd(kwd, ggp));
145
146    let union = j(Token::CurlOpen)
147        .rewind()
148        .ignore_then(ggp.clone())
149        .map_with_span(spanned)
150        .then(
151            sparql_kwd(SparqlKeyword::Union)
152                .then(ggp.map_with_span(spanned))
153                .repeated(),
154        )
155        .map(|(start, rest)| GroupGraphPatternSub::Union(start, rest));
156
157    // TODO add the others
158    //
159
160    next_check.ignore_then(trip.or(kwd).or(union).labelled("group_graph_pattern_sub"))
161}

163fn expect_it(
164    token: Token,
165    st: &'static str,
166) -> impl Parser<PToken, Token, Error = Simple<PToken>> + Clone {
167    j(token.clone()).or(not(token.clone())
168        .map(|x| x.0)
169        .try_map(move |x: Token, span| {
170            debug!("unexpected token: {} didn't expect {}", st, x);
171            Err(Simple::expected_input_found(
172                span,
173                [Some(PToken(token.clone(), 0))],
174                Some(PToken(x.clone(), 0)),
175            ))
176        }))
177}

180fn group_graph_pattern<'a, T: Parser<PToken, SubSelect, Error = Simple<PToken>> + Clone + 'a>(
181    select: T,
182    ctx: Ctx<'a>,
183) -> impl Parser<PToken, GroupGraphPattern, Error = Simple<PToken>> + Clone + use<'a, T> {
184    let s = select.clone();
185    recursive(|ggp| {
186        let select = s
187            .clone()
188            .map(Box::from)
189            .map(GroupGraphPattern::SubSelect)
190            .labelled("sub_select");
191
192        let gg = group_graph_pattern_sub(ggp, ctx)
193            .map_with_span(spanned)
194            .repeated()
195            .map(GroupGraphPattern::GroupGraph);
196
197        let close = expect_it(Token::CurlClose, "close").labelled("CurlClose");
198        let open = expect_it(Token::CurlOpen, "open").labelled("CurlOpen");
199
200        open.ignore_then(gg.or(select)).then_ignore(close)
201    })
202}

217    let num = select!(
218        PToken(Token::Number(x), _) => x,
219    )
220    .map_with_span(spanned);
221
222    let limit_offset = sparql_kwd(SparqlKeyword::Limit)
223        .or(sparql_kwd(SparqlKeyword::Offset))
224        .then(num)
225        .map(|(kwd, num)| Modifier::LimitOffset(kwd, num));
226    limit_offset
227}

229pub fn query<'a>(
230    base: swls_core::lsp_types::Url,
231    ctx: Ctx<'a>,
232) -> impl Parser<PToken, Query, Error = Simple<PToken>> + Clone + use<'a> {
233    let prologues = prologue().map_with_span(spanned).repeated().map(|xs| {
234        let mut base = None;
235        let mut prefixes = vec![];
236        xs.into_iter().for_each(|Spanned(x, span)| match x {
237            Prologue::Base { token, iri } => base = Some(Spanned(Base { token, iri }, span)),
238            Prologue::Prefix(prefix) => prefixes.push(Spanned(prefix, span)),
239        });
240        (base, prefixes)
241    });
242    let kwds = select_clause().map(QueryClause::Select);
243    let datasets = dataset_clause().map_with_span(spanned).repeated();
244    let where_clause = where_clause(sub_select(ctx), ctx).map_with_span(spanned);
245    let modifiers = modifier().map_with_span(spanned).repeated();
246
247    prologues
248        .then(kwds)
249        .then(datasets)
250        .then(where_clause)
251        .then(modifiers)
252        .map(
253            move |(((((base_statement, prefixes), kwds), datasets), where_clause), modifier)| {
254                Query {
255                    base_statement,
256                    prefixes,
257                    base: base.clone(),
258                    modifier,
259                    where_clause,
260                    datasets,
261                    kwds,
262                }
263            },
264        )
265}

267pub fn parse(
268    source: &str,
269    base: swls_core::lsp_types::Url,
270    tokens: Vec<Spanned<Token>>,
271    ctx: Ctx<'_>,
272) -> (Spanned<Query>, Vec<Simple<PToken>>) {
273    let len = source.len();
274    let stream = chumsky::Stream::from_iter(
275        0..len,
276        tokens
277            .into_iter()
278            .enumerate()
279            .filter(|(_, x)| !x.is_comment())
280            .map(|(i, t)| t.map(|x| PToken(x, i)))
281            .map(|Spanned(x, s)| (x, s)),
282    );
283
284    let parser = query(base, ctx)
285        .map_with_span(spanned)
286        .then_ignore(end().recover_with(skip_then_retry_until([])));
287    let (json, json_errors) = parser.parse_recovery(stream);
288
289    (
290        json.unwrap_or(Spanned(Query::default(), 0..source.len())),
291        json_errors,
292    )
293}
