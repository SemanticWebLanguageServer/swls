Struct Color 

Source 
pub struct Color { /* private fields */ }
Expand description

Color represents a color which can be set to things like Border, Padding and Margin.

§Example

use tabled::{settings::Color, Table};

let data = [
    (0u8, "Hello"),
    (1u8, "World"),
];

let table = Table::new(data)
    .with(Color::BG_BLUE)
    .to_string();

println!("{}", table);
