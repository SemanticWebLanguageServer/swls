use std::str::FromStr;
use resiter::oks::GetOks;

let res:Vec<usize> = ["1", "2", "3", "a", "4", "5"]
    .iter()
    .map(|e| usize::from_str(e))
    .oks()
    .collect();

assert_eq!(
    res,
    vec![1,2,3,4,5]
);
